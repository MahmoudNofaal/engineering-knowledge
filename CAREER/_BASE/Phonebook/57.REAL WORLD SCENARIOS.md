## 57.01 — Building a Complete Feature (Order Cancellation)

**Domain layer**
```csharp
// Domain/Entities/Order.cs
public class Order : AggregateRoot
{
    public OrderId Id { get; private set; }
    public CustomerId CustomerId { get; private set; }
    public OrderStatus Status { get; private set; }
    public DateTime CreatedAt { get; private set; }
    public DateTime? CancelledAt { get; private set; }
    public string? CancellationReason { get; private set; }
    
    public static Order Create(CustomerId customerId) { ... }
    
    public void Cancel(string reason)
    {
        if (Status == OrderStatus.Cancelled)
            throw new DomainException("Order is already cancelled");
        
        if (Status == OrderStatus.Shipped || Status == OrderStatus.Delivered)
            throw new DomainException("Cannot cancel a shipped or delivered order");
        
        Status = OrderStatus.Cancelled;
        CancelledAt = DateTime.UtcNow;
        CancellationReason = reason;
        
        Raise(new OrderCancelledEvent(Id, CustomerId, reason, DateTime.UtcNow));
    }
}

// Domain/Events/OrderCancelledEvent.cs
public record OrderCancelledEvent(
    OrderId OrderId,
    CustomerId CustomerId,
    string Reason,
    DateTime CancelledAt) : DomainEvent;
```

**Application layer**
```csharp
// Application/Commands/CancelOrderCommand.cs
public record CancelOrderCommand(int OrderId, string Reason, int RequestingUserId)
    : IRequest<CancelOrderResult>;

public record CancelOrderResult(int OrderId, string Status, DateTime CancelledAt);

// Application/Commands/CancelOrderHandler.cs
public class CancelOrderHandler : IRequestHandler<CancelOrderCommand, CancelOrderResult>
{
    private readonly IOrderRepository _orders;
    private readonly IUnitOfWork _uow;
    
    public async Task<CancelOrderResult> Handle(
        CancelOrderCommand cmd, CancellationToken ct)
    {
        var order = await _orders.GetByIdAsync(new OrderId(cmd.OrderId), ct)
            ?? throw new NotFoundException($"Order {cmd.OrderId} not found");
        
        if (order.CustomerId.Value != cmd.RequestingUserId)
            throw new ForbiddenException("You can only cancel your own orders");
        
        order.Cancel(cmd.Reason);
        await _uow.SaveChangesAsync(ct);
        
        return new CancelOrderResult(cmd.OrderId, order.Status.ToString(), order.CancelledAt!.Value);
    }
}

// Application/Commands/CancelOrderValidator.cs
public class CancelOrderValidator : AbstractValidator<CancelOrderCommand>
{
    public CancelOrderValidator()
    {
        RuleFor(x => x.OrderId).GreaterThan(0);
        RuleFor(x => x.Reason)
            .NotEmpty().WithMessage("Cancellation reason is required")
            .MaximumLength(500);
    }
}
```

group.MapDelete("{orderId:int}", CancelOrder)
    .WithName("CancelOrder")
    .Produces<CancelOrderResult>(200)
    .ProducesProblem(404)
    .ProducesProblem(403)
    .ProducesValidationProblem()
    .RequireAuthorization();

private static async Task<IResult> CancelOrder(
    int orderId,
    CancelOrderRequest request,
    ISender mediator,
    IValidator<CancelOrderCommand> validator,
    ClaimsPrincipal user,
    CancellationToken ct)
{
    var userId = int.Parse(user.FindFirstValue(ClaimTypes.NameIdentifier)!);
    var command = new CancelOrderCommand(orderId, request.Reason, userId);
    
    var validation = await validator.ValidateAsync(command, ct);
    if (!validation.IsValid)
        return Results.ValidationProblem(validation.ToDictionary());
    
    try
    {
        var result = await mediator.Send(command, ct);
        return Results.Ok(result);
    }
    catch (NotFoundException ex)
    {
        return Results.NotFound(new { ex.Message });
    }
    catch (ForbiddenException)
    {
        return Results.Forbid();
    }
    catch (DomainException ex)
    {
        return Results.UnprocessableEntity(new { ex.Message });
    }
}

public class CancelOrderHandlerTests

    private readonly IUnitOfWork _uow = Substitute.For<IUnitOfWork>();

        // Act
        var result = await _sut.Handle(command, CancellationToken.None);
        
        // Assert
        result.Status.Should().Be("Cancelled");
        await _uow.Received(1).SaveChangesAsync(Arg.Any<CancellationToken>());

## 57.02 — Debugging a Real Issue (Memory Leak)

**The scenario**
Production API: memory grows steadily over 3 hours, then OOM crash. No memory pressure visible. GC.Gen2 count climbing steadily.

**Investigation process**

```
Step 1: Reproduce (or capture from production)
dotnet gcdump collect --process-id <PID>

Step 2: Open in VS 2022 or PerfView
Filter by type → look for unexpected large counts

Step 3: Findings:
  System.Collections.Generic.Dictionary<String, Object> — 45,000 instances (!!)
  MyApp.Services.UserSession — 45,000 instances

Step 4: Who is holding references?
In dotnet dump:
> gcroot <address of one UserSession instance>
Output:
  → ConcurrentDictionary<string, UserSession> (static field)
  → UserSessionManager._sessions
  → Thread → ...

Step 5: The bug
private static readonly ConcurrentDictionary<string, UserSession> _sessions = new();

public void CreateSession(string token, UserSession session)
    => _sessions.TryAdd(token, session);

// Sessions are never removed! They accumulate indefinitely.
// Expired sessions fill the dictionary forever.

Step 6: Fix
// Option A: TTL-based eviction
public class UserSessionManager : BackgroundService
{
    private readonly ConcurrentDictionary<string, (UserSession Session, DateTime Expiry)> _sessions = new();
    
    public void CreateSession(string token, UserSession session, TimeSpan ttl)
        => _sessions.TryAdd(token, (session, DateTime.UtcNow.Add(ttl)));
    
    public UserSession? GetSession(string token)
    {
        if (!_sessions.TryGetValue(token, out var entry)) return null;
        if (entry.Expiry < DateTime.UtcNow) { _sessions.TryRemove(token, out _); return null; }
        return entry.Session;
    }
    
    // Background cleanup
    protected override async Task ExecuteAsync(CancellationToken ct)
    {
        using var timer = new PeriodicTimer(TimeSpan.FromMinutes(5));
        while (await timer.WaitForNextTickAsync(ct))
        {
            var expired = _sessions.Where(kvp => kvp.Value.Expiry < DateTime.UtcNow).Select(kvp => kvp.Key);
            foreach (var key in expired)
                _sessions.TryRemove(key, out _);
        }
    }
}

// Option B (better): Use IMemoryCache with expiry
// IMemoryCache handles eviction automatically
```
