## 38.01 — ETL vs. ELT

**ETL (Extract, Transform, Load)**
```
Source → Extract → Transform (in pipeline) → Load into warehouse
```
Transform happens before loading. Good when the warehouse is expensive or transformation is complex.

**ELT (Extract, Load, Transform)**
```
Source → Extract → Load raw → Transform in warehouse
```
Load raw data, transform in the warehouse using SQL. Good when the warehouse is powerful (Snowflake, BigQuery). Modern approach.

**In .NET for ETL/ELT pipelines**
```csharp
// Simple ETL with Channel pipeline
public class SalesDataPipeline
{
    public async Task RunAsync(CancellationToken ct)
    {
        var extractChannel = Channel.CreateBounded<RawSaleRecord>(1000);
        var transformChannel = Channel.CreateBounded<TransformedSale>(1000);
        
        await Task.WhenAll(
            ExtractFromCsvAsync(extractChannel.Writer, ct),
            TransformAsync(extractChannel.Reader, transformChannel.Writer, ct),
            LoadToWarehouseAsync(transformChannel.Reader, ct)
        );
    }
    
    private async Task ExtractFromCsvAsync(ChannelWriter<RawSaleRecord> writer, CancellationToken ct)
    {
        // Stream CSV without loading entire file into memory
        await foreach (var line in ReadCsvLinesAsync("sales.csv", ct))
        {
            var record = ParseCsvLine(line);
            await writer.WriteAsync(record, ct);
        }
        writer.Complete();
    }
    
    private async Task TransformAsync(
        ChannelReader<RawSaleRecord> reader,
        ChannelWriter<TransformedSale> writer,
        CancellationToken ct)
    {
        await foreach (var raw in reader.ReadAllAsync(ct))
        {
            var transformed = new TransformedSale
            {
                SaleDate = DateOnly.Parse(raw.DateString),
                Amount = decimal.Parse(raw.AmountString),
                Region = NormalizeRegion(raw.Region),
                ProductCategory = MapCategory(raw.ProductCode)
            };
            await writer.WriteAsync(transformed, ct);
        }
        writer.Complete();
    }
    
    private async Task LoadToWarehouseAsync(ChannelReader<TransformedSale> reader, CancellationToken ct)
    {
        // Batch bulk insert for efficiency
        var batch = new List<TransformedSale>(1000);
        
        await foreach (var record in reader.ReadAllAsync(ct))
        {
            batch.Add(record);
            if (batch.Count >= 1000)
            {
                await BulkInsertAsync(batch, ct);
                batch.Clear();
            }
        }
        
        if (batch.Any())
            await BulkInsertAsync(batch, ct);
    }
}
```
