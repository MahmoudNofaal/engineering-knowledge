## 13.01 — Window Functions

**What it is**
Functions that perform calculations across a set of rows related to the current row, without collapsing rows like GROUP BY.

**Syntax**
```sql
function_name() OVER (
    PARTITION BY column1, column2   -- optional: divide into groups
    ORDER BY column3                -- optional: order within group
    ROWS/RANGE BETWEEN ... AND ...  -- optional: frame definition
)
```

**Ranking functions**
```sql
SELECT
    EmployeeId,
    Name,
    Salary,
    DepartmentId,
    ROW_NUMBER() OVER (PARTITION BY DepartmentId ORDER BY Salary DESC) AS RowNum,
    RANK()       OVER (PARTITION BY DepartmentId ORDER BY Salary DESC) AS Rnk,
    DENSE_RANK() OVER (PARTITION BY DepartmentId ORDER BY Salary DESC) AS DenseRnk,
    NTILE(4)     OVER (PARTITION BY DepartmentId ORDER BY Salary DESC) AS Quartile
FROM Employees;

-- ROW_NUMBER: 1, 2, 3, 4, 5 (always unique)
-- RANK:       1, 2, 2, 4, 5 (ties get same rank, gap after)
-- DENSE_RANK: 1, 2, 2, 3, 4 (ties get same rank, no gap)
-- NTILE(4):   divides into 4 groups, assigns 1-4
```

**Aggregate window functions**
```sql
SELECT
    OrderDate,
    Amount,
    SUM(Amount) OVER (ORDER BY OrderDate) AS RunningTotal,
    SUM(Amount) OVER (PARTITION BY MONTH(OrderDate)) AS MonthTotal,
    AVG(Amount) OVER (ORDER BY OrderDate ROWS BETWEEN 6 PRECEDING AND CURRENT ROW) AS SevenDayAvg,
    COUNT(*)    OVER () AS TotalOrders
FROM Orders;
```

**LAG and LEAD — accessing adjacent rows**
```sql
SELECT
    OrderDate,
    Amount,
    LAG(Amount, 1, 0) OVER (ORDER BY OrderDate) AS PreviousAmount,
    LEAD(Amount, 1, 0) OVER (ORDER BY OrderDate) AS NextAmount,
    Amount - LAG(Amount) OVER (ORDER BY OrderDate) AS ChangeFromPrevious
FROM Orders;
```

**FIRST_VALUE and LAST_VALUE**
```sql
SELECT
    ProductId,
    Category,
    Price,
    FIRST_VALUE(Price) OVER (PARTITION BY Category ORDER BY Price) AS MinPriceInCategory,
    LAST_VALUE(Price) OVER (
        PARTITION BY Category ORDER BY Price
        ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING  -- Important!
    ) AS MaxPriceInCategory
FROM Products;
```

**Practical example: Top-N per group**
```sql
-- Get top 3 orders per customer
WITH RankedOrders AS (
    SELECT
        OrderId,
        CustomerId,
        Total,
        ROW_NUMBER() OVER (PARTITION BY CustomerId ORDER BY Total DESC) AS Rn
    FROM Orders
)
SELECT * FROM RankedOrders WHERE Rn <= 3;
```

## 13.02 — CTEs (Common Table Expressions)

**What it is**
Named temporary result sets that you can reference within a SELECT, INSERT, UPDATE, or DELETE statement. Improve readability over subqueries.

**Basic CTE**
```sql
WITH ActiveCustomers AS (
    SELECT CustomerId, Name, Email
    FROM Customers
    WHERE IsActive = 1
),
RecentOrders AS (
    SELECT CustomerId, COUNT(*) AS OrderCount, SUM(Total) AS TotalSpent
    FROM Orders
    WHERE OrderDate >= DATEADD(MONTH, -3, GETDATE())
    GROUP BY CustomerId
)
SELECT
    ac.Name,
    ac.Email,
    COALESCE(ro.OrderCount, 0) AS RecentOrders,
    COALESCE(ro.TotalSpent, 0) AS RecentSpend
FROM ActiveCustomers ac
LEFT JOIN RecentOrders ro ON ac.CustomerId = ro.CustomerId;
```

**Recursive CTE**
Used for hierarchical data (org charts, category trees, bill of materials).

```sql
-- Org chart: find all employees under a manager
WITH OrgHierarchy AS (
    -- Anchor: start with the root manager
    SELECT EmployeeId, Name, ManagerId, 0 AS Level
    FROM Employees
    WHERE ManagerId IS NULL  -- Top level
    
    UNION ALL
    
    -- Recursive: join back to find direct reports
    SELECT e.EmployeeId, e.Name, e.ManagerId, oh.Level + 1
    FROM Employees e
    INNER JOIN OrgHierarchy oh ON e.ManagerId = oh.EmployeeId
)
SELECT EmployeeId, Name, Level, REPLICATE('  ', Level) + Name AS Hierarchy
FROM OrgHierarchy
ORDER BY Level, Name;

-- Category tree: find all descendants of a category
WITH CategoryTree AS (
    SELECT CategoryId, Name, ParentCategoryId, 0 AS Depth
    FROM Categories
    WHERE CategoryId = @rootCategoryId
    
    UNION ALL
    
    SELECT c.CategoryId, c.Name, c.ParentCategoryId, ct.Depth + 1
    FROM Categories c
    INNER JOIN CategoryTree ct ON c.ParentCategoryId = ct.CategoryId
)
SELECT * FROM CategoryTree;
```

## 13.03 — Advanced Joins

**Self Join**
```sql
-- Find employees and their managers
SELECT
    e.Name AS Employee,
    m.Name AS Manager
FROM Employees e
LEFT JOIN Employees m ON e.ManagerId = m.EmployeeId;
```

**Cross Join**
```sql
-- Generate all combinations
SELECT s.Size, c.Color
FROM Sizes s
CROSS JOIN Colors c;
-- 4 sizes × 8 colors = 32 rows
```

**FULL OUTER JOIN**
```sql
-- All rows from both tables
SELECT
    c.Name AS Customer,
    o.OrderId
FROM Customers c
FULL OUTER JOIN Orders o ON c.CustomerId = o.CustomerId;
-- Includes customers with no orders AND orders with no customer
```

**EXCEPT and INTERSECT**
```sql
-- Customers who have NOT placed an order in 2026
SELECT CustomerId FROM Customers
EXCEPT
SELECT DISTINCT CustomerId FROM Orders WHERE YEAR(OrderDate) = 2026;

-- Products in both the featured list and the sale list
SELECT ProductId FROM FeaturedProducts
INTERSECT
SELECT ProductId FROM SaleProducts;
```

## 13.04 — JSON in SQL Server

**Storing and querying JSON**
```sql
-- Store JSON in a column
CREATE TABLE Events (
    EventId INT PRIMARY KEY,
    EventType VARCHAR(50),
    EventData NVARCHAR(MAX)  -- JSON stored as string
);

INSERT INTO Events VALUES (1, 'OrderCreated', '{"orderId": 42, "customerId": 7, "items": [{"productId": 1, "qty": 2}]}');

-- Query JSON values
SELECT
    EventId,
    JSON_VALUE(EventData, '$.orderId') AS OrderId,
    JSON_VALUE(EventData, '$.customerId') AS CustomerId,
    JSON_QUERY(EventData, '$.items') AS ItemsJson  -- Returns array as string
FROM Events
WHERE JSON_VALUE(EventData, '$.customerId') = '7';

-- Parse JSON array into rows
SELECT item.*
FROM Events e
CROSS APPLY OPENJSON(e.EventData, '$.items')
WITH (
    productId INT '$.productId',
    qty INT '$.qty'
) AS item
WHERE e.EventId = 1;

-- FOR JSON — return query results as JSON
SELECT OrderId, CustomerId, Total
FROM Orders
FOR JSON PATH, ROOT('orders');
```

## 13.05 — Temporal Tables (System-Versioned)

**What it is**
Tables that automatically track history — every row has a valid-from and valid-to timestamp, maintained by SQL Server.

```sql
-- Create temporal table
CREATE TABLE Products (
    ProductId INT PRIMARY KEY,
    Name NVARCHAR(100),
    Price DECIMAL(10, 2),
    ValidFrom DATETIME2 GENERATED ALWAYS AS ROW START NOT NULL,
    ValidTo DATETIME2 GENERATED ALWAYS AS ROW END NOT NULL,
    PERIOD FOR SYSTEM_TIME (ValidFrom, ValidTo)
) WITH (SYSTEM_VERSIONING = ON (HISTORY_TABLE = dbo.ProductsHistory));

-- Changes are tracked automatically:
UPDATE Products SET Price = 29.99 WHERE ProductId = 1;
-- History table now has the old row with ValidTo = update time

-- Query as of a point in time
SELECT * FROM Products
FOR SYSTEM_TIME AS OF '2026-01-01 00:00:00'
WHERE ProductId = 1;

-- Query history for a range
SELECT * FROM Products
FOR SYSTEM_TIME BETWEEN '2025-01-01' AND '2026-01-01'
WHERE ProductId = 1
ORDER BY ValidFrom;
```
