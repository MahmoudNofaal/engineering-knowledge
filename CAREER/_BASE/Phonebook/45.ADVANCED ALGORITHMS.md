## 45.01 — Sorting Algorithms (Implementation + Analysis)

**Merge Sort — O(n log n), stable, O(n) space**
```csharp
void MergeSort(int[] arr, int left, int right)
{
    if (left >= right) return;
    
    int mid = left + (right - left) / 2;
    MergeSort(arr, left, mid);
    MergeSort(arr, mid + 1, right);
    Merge(arr, left, mid, right);
}

void Merge(int[] arr, int left, int mid, int right)
{
    int[] temp = new int[right - left + 1];
    int i = left, j = mid + 1, k = 0;
    
    while (i <= mid && j <= right)
        temp[k++] = arr[i] <= arr[j] ? arr[i++] : arr[j++];
    
    while (i <= mid) temp[k++] = arr[i++];
    while (j <= right) temp[k++] = arr[j++];
    
    Array.Copy(temp, 0, arr, left, temp.Length);
}
```

**Quick Sort — O(n log n) average, O(n²) worst, O(log n) space**
```csharp
void QuickSort(int[] arr, int low, int high)
{
    if (low >= high) return;
    
    int pivot = Partition(arr, low, high);
    QuickSort(arr, low, pivot - 1);
    QuickSort(arr, pivot + 1, high);
}

int Partition(int[] arr, int low, int high)
{
    // Median of three pivot to avoid worst case
    int mid = low + (high - low) / 2;
    if (arr[mid] < arr[low]) Swap(arr, low, mid);
    if (arr[high] < arr[low]) Swap(arr, low, high);
    if (arr[mid] < arr[high]) Swap(arr, mid, high);
    
    int pivot = arr[high];
    int i = low - 1;
    
    for (int j = low; j < high; j++)
        if (arr[j] <= pivot)
            Swap(arr, ++i, j);
    
    Swap(arr, i + 1, high);
    return i + 1;
}

void Swap(int[] arr, int i, int j) => (arr[i], arr[j]) = (arr[j], arr[i]);
```

**Counting Sort — O(n + k), stable**
```csharp
int[] CountingSort(int[] arr, int maxVal)
{
    var count = new int[maxVal + 1];
    foreach (int x in arr) count[x]++;
    
    // Prefix sums
    for (int i = 1; i <= maxVal; i++) count[i] += count[i - 1];
    
    var output = new int[arr.Length];
    for (int i = arr.Length - 1; i >= 0; i--)  // Iterate backward for stability
    {
        output[--count[arr[i]]] = arr[i];
    }
    return output;
}
```

**Radix Sort — O(d × (n + k)), non-comparison sort**
```csharp
void RadixSort(int[] arr)
{
    int max = arr.Max();
    for (int exp = 1; max / exp > 0; exp *= 10)
        CountingByDigit(arr, exp);
}

void CountingByDigit(int[] arr, int exp)
{
    var output = new int[arr.Length];
    var count = new int[10];
    
    foreach (int x in arr) count[(x / exp) % 10]++;
    for (int i = 1; i < 10; i++) count[i] += count[i - 1];
    for (int i = arr.Length - 1; i >= 0; i--)
        output[--count[(arr[i] / exp) % 10]] = arr[i];
    
    output.CopyTo(arr, 0);
}
```

## 45.02 — Advanced Graph Algorithms

**Bellman-Ford (handles negative weights)**
```csharp
int[] BellmanFord(int n, int[][] edges, int source)
{
    var dist = Enumerable.Repeat(int.MaxValue, n).ToArray();
    dist[source] = 0;
    
    // Relax edges n-1 times
    for (int i = 0; i < n - 1; i++)
    {
        foreach (var edge in edges)
        {
            int u = edge[0], v = edge[1], w = edge[2];
            if (dist[u] != int.MaxValue && dist[u] + w < dist[v])
                dist[v] = dist[u] + w;
        }
    }
    
    // Check for negative cycles
    foreach (var edge in edges)
    {
        int u = edge[0], v = edge[1], w = edge[2];
        if (dist[u] != int.MaxValue && dist[u] + w < dist[v])
            throw new InvalidOperationException("Negative cycle detected");
    }
    
    return dist;
}
```

**Floyd-Warshall (all-pairs shortest path)**
```csharp
int[,] FloydWarshall(int n, int[,] graph)
{
    var dist = (int[,])graph.Clone();
    
    for (int k = 0; k < n; k++)
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                if (dist[i, k] != int.MaxValue && dist[k, j] != int.MaxValue)
                    dist[i, j] = Math.Min(dist[i, j], dist[i, k] + dist[k, j]);
    
    return dist;
}
```

**Tarjan's SCC (Strongly Connected Components)**
```csharp
public class TarjanSCC
{
    private int _timer = 0;
    private readonly Stack<int> _stack = new();
    private readonly int[] _disc, _low;
    private readonly bool[] _inStack;
    private readonly List<List<int>> _sccs = new();
    private readonly List<int>[] _adj;
    
    public TarjanSCC(int n, List<int>[] adj)
    {
        _adj = adj;
        _disc = Enumerable.Repeat(-1, n).ToArray();
        _low = new int[n];
        _inStack = new bool[n];
    }
    
    public List<List<int>> FindSCCs(int n)
    {
        for (int i = 0; i < n; i++)
            if (_disc[i] == -1)
                DFS(i);
        return _sccs;
    }
    
    private void DFS(int u)
    {
        _disc[u] = _low[u] = _timer++;
        _stack.Push(u);
        _inStack[u] = true;
        
        foreach (int v in _adj[u])
        {
            if (_disc[v] == -1)
            {
                DFS(v);
                _low[u] = Math.Min(_low[u], _low[v]);
            }
            else if (_inStack[v])
            {
                _low[u] = Math.Min(_low[u], _disc[v]);
            }
        }
        
        if (_low[u] == _disc[u])
        {
            var scc = new List<int>();
            while (true)
            {
                var v = _stack.Pop();
                _inStack[v] = false;
                scc.Add(v);
                if (v == u) break;
            }
            _sccs.Add(scc);
        }
    }
}
```

## 45.03 — String Algorithms

**KMP (Knuth-Morris-Pratt) — O(n + m) string search**
```csharp
int[] KMPSearch(string text, string pattern)
{
    var matches = new List<int>();
    var lps = BuildLPS(pattern);
    
    int i = 0, j = 0;  // i = text, j = pattern
    while (i < text.Length)
    {
        if (text[i] == pattern[j])
        {
            i++; j++;
            if (j == pattern.Length)
            {
                matches.Add(i - j);
                j = lps[j - 1];
            }
        }
        else if (j > 0)
            j = lps[j - 1];
        else
            i++;
    }
    return matches.ToArray();
}

int[] BuildLPS(string pattern)
{
    var lps = new int[pattern.Length];
    int len = 0, i = 1;
    
    while (i < pattern.Length)
    {
        if (pattern[i] == pattern[len])
            lps[i++] = ++len;
        else if (len > 0)
            len = lps[len - 1];
        else
            lps[i++] = 0;
    }
    return lps;
}
```

**Rabin-Karp — O(n + m) average, rolling hash**
```csharp
IList<int> RabinKarp(string text, string pattern)
{
    int n = text.Length, m = pattern.Length;
    const int Base = 31, Mod = 1_000_000_007;
    var matches = new List<int>();
    
    long patHash = 0, textHash = 0, power = 1;
    
    for (int i = m - 1; i >= 0; i--)
    {
        patHash = (patHash + (pattern[i] - 'a' + 1) * power) % Mod;
        textHash = (textHash + (text[i] - 'a' + 1) * power) % Mod;
        if (i > 0) power = power * Base % Mod;
    }
    
    for (int i = 0; i <= n - m; i++)
    {
        if (textHash == patHash && text.AsSpan(i, m).SequenceEqual(pattern))
            matches.Add(i);
        
        if (i < n - m)
        {
            textHash = (textHash - (text[i] - 'a' + 1) * power % Mod + Mod) % Mod;
            textHash = (textHash * Base + (text[i + m] - 'a' + 1)) % Mod;
        }
    }
    return matches;
}
```
