## 96.01 — Specification Pattern

**What it is**
Encapsulate business rules as composable objects. Combine with AND, OR, NOT.

```csharp
// Interface
public interface ISpecification<T>
{
    bool IsSatisfiedBy(T candidate);
    ISpecification<T> And(ISpecification<T> other);
    ISpecification<T> Or(ISpecification<T> other);
    ISpecification<T> Not();
}

// Base implementation
public abstract class Specification<T> : ISpecification<T>
{
    public abstract bool IsSatisfiedBy(T candidate);
    
    public ISpecification<T> And(ISpecification<T> other)
        => new AndSpecification<T>(this, other);
    
    public ISpecification<T> Or(ISpecification<T> other)
        => new OrSpecification<T>(this, other);
    
    public ISpecification<T> Not()
        => new NotSpecification<T>(this);
}

public class AndSpecification<T> : Specification<T>
{
    private readonly ISpecification<T> _left, _right;
    public AndSpecification(ISpecification<T> left, ISpecification<T> right)
    { _left = left; _right = right; }
    public override bool IsSatisfiedBy(T candidate)
        => _left.IsSatisfiedBy(candidate) && _right.IsSatisfiedBy(candidate);
}

public class OrSpecification<T> : Specification<T>
{
    private readonly ISpecification<T> _left, _right;
    public OrSpecification(ISpecification<T> left, ISpecification<T> right)
    { _left = left; _right = right; }
    public override bool IsSatisfiedBy(T candidate)
        => _left.IsSatisfiedBy(candidate) || _right.IsSatisfiedBy(candidate);
}

public class NotSpecification<T> : Specification<T>
{
    private readonly ISpecification<T> _inner;
    public NotSpecification(ISpecification<T> inner) => _inner = inner;
    public override bool IsSatisfiedBy(T candidate) => !_inner.IsSatisfiedBy(candidate);
}

// Domain specifications
public class OrderEligibleForDiscount : Specification<Order>
{
    public override bool IsSatisfiedBy(Order order)
        => order.Total > 100 && order.Items.Count >= 3;
}

public class OrderIsFromVIPCustomer : Specification<Order>
{
    private readonly ISet<int> _vipCustomerIds;
    public OrderIsFromVIPCustomer(ISet<int> vipIds) => _vipCustomerIds = vipIds;
    public override bool IsSatisfiedBy(Order order)
        => _vipCustomerIds.Contains(order.CustomerId);
}

public class OrderIsRecent : Specification<Order>
{
    public override bool IsSatisfiedBy(Order order)
        => order.CreatedAt > DateTime.UtcNow.AddDays(-7);
}

// Usage — composable business rules
var discountSpec = new OrderEligibleForDiscount()
    .Or(new OrderIsFromVIPCustomer(vipIds))
    .And(new OrderIsRecent());

var eligibleOrders = orders.Where(o => discountSpec.IsSatisfiedBy(o)).ToList();

// Translate to EF Core queries (for database-level filtering)
public abstract class ExpressionSpecification<T> : Specification<T>
{
    public abstract Expression<Func<T, bool>> ToExpression();
    
    public override bool IsSatisfiedBy(T candidate)
        => ToExpression().Compile()(candidate);
}

public class LargeOrderSpec : ExpressionSpecification<Order>
{
    public override Expression<Func<Order, bool>> ToExpression()
        => order => order.Total > 1000;
}

// Apply to EF Core query
var spec = new LargeOrderSpec();
var orders = await _db.Orders
    .Where(spec.ToExpression())  // Translated to SQL
    .ToListAsync();
```

## 96.02 — Outbox Pattern (Complete)

**The problem**
Save data to DB AND publish event — atomically. If DB save succeeds but event publish fails, the event is lost.

**The solution**
Save the event INTO the database (as part of the same transaction). A background worker reads and publishes the events.

```csharp
// Outbox message table
public class OutboxMessage
{
    public Guid Id { get; set; } = Guid.NewGuid();
    public string Type { get; set; } = null!;         // Event class name
    public string Payload { get; set; } = null!;       // Serialized event
    public DateTime CreatedAt { get; set; } = DateTime.UtcNow;
    public DateTime? ProcessedAt { get; set; }          // Null = not yet processed
    public int RetryCount { get; set; } = 0;
    public string? Error { get; set; }
}

// EF Core config
builder.Entity<OutboxMessage>(b =>
{
    b.ToTable("OutboxMessages");
    b.HasKey(m => m.Id);
    b.HasIndex(m => m.ProcessedAt);  // For efficient pending query
    b.Property(m => m.Payload).HasColumnType("nvarchar(max)");
});

// Publish within the SAME transaction
public class UnitOfWork : IUnitOfWork
{
    private readonly AppDbContext _db;
    
    public void Publish(IEnumerable<DomainEvent> events)
    {
        foreach (var @event in events)
        {
            _db.OutboxMessages.Add(new OutboxMessage
            {
                Type = @event.GetType().FullName!,
                Payload = JsonSerializer.Serialize(@event, @event.GetType())
            });
        }
    }
    
    public async Task SaveChangesAsync(CancellationToken ct)
    {
        // Collect domain events from aggregates
        var aggregates = _db.ChangeTracker.Entries<AggregateRoot>()
            .Select(e => e.Entity)
            .Where(a => a.GetEvents().Any())
            .ToList();
        
        Publish(aggregates.SelectMany(a => a.GetEvents()));
        aggregates.ForEach(a => a.ClearEvents());
        
        await _db.SaveChangesAsync(ct);  // Saves domain data + outbox messages atomically
    }
}

// Background worker — reads and publishes outbox messages
public class OutboxProcessor : BackgroundService
{
    private readonly IServiceScopeFactory _scopeFactory;
    private readonly IMessagePublisher _publisher;
    
    protected override async Task ExecuteAsync(CancellationToken ct)
    {
        using var timer = new PeriodicTimer(TimeSpan.FromSeconds(5));
        
        while (await timer.WaitForNextTickAsync(ct))
        {
            await ProcessPendingMessagesAsync(ct);
        }
    }
    
    private async Task ProcessPendingMessagesAsync(CancellationToken ct)
    {
        await using var scope = _scopeFactory.CreateAsyncScope();
        var db = scope.ServiceProvider.GetRequiredService<AppDbContext>();
        
        var pending = await db.OutboxMessages
            .Where(m => m.ProcessedAt == null && m.RetryCount < 5)
            .OrderBy(m => m.CreatedAt)
            .Take(20)
            .ToListAsync(ct);
        
        foreach (var message in pending)
        {
            try
            {
                // Deserialize event
                var eventType = Type.GetType(message.Type)!;
                var @event = JsonSerializer.Deserialize(message.Payload, eventType)!;
                
                // Publish to message bus
                await _publisher.PublishAsync(@event, ct);
                
                // Mark as processed
                message.ProcessedAt = DateTime.UtcNow;
            }
            catch (Exception ex)
            {
                message.RetryCount++;
                message.Error = ex.Message;
                // Will retry next cycle up to MaxRetryCount times
            }
        }
        
        await db.SaveChangesAsync(ct);
    }
}
```

## 96.03 — Saga Pattern (Orchestration)

**What it is**
A sequence of local transactions coordinated by a central orchestrator. Each step has a compensating action for rollback.

```csharp
// Order processing saga
public class PlaceOrderSaga : ISaga
{
    public SagaState State { get; private set; } = SagaState.NotStarted;
    private int _orderId;
    private int _paymentId;
    private int _reservationId;
    
    public async Task ExecuteAsync(PlaceOrderCommand cmd, CancellationToken ct)
    {
        try
        {
            // Step 1: Create order
            State = SagaState.CreatingOrder;
            _orderId = await _orderService.CreateAsync(cmd, ct);
            
            // Step 2: Reserve inventory
            State = SagaState.ReservingInventory;
            _reservationId = await _inventoryService.ReserveAsync(_orderId, cmd.Items, ct);
            
            // Step 3: Process payment
            State = SagaState.ProcessingPayment;
            _paymentId = await _paymentService.ChargeAsync(_orderId, cmd.Amount, ct);
            
            // Step 4: Confirm order
            State = SagaState.ConfirmingOrder;
            await _orderService.ConfirmAsync(_orderId, ct);
            
            State = SagaState.Completed;
        }
        catch (InventoryException ex)
        {
            // Compensate: cancel order (payment not yet processed)
            await CompensateAsync(ct);
            throw new SagaException("Failed at inventory reservation", ex);
        }
        catch (PaymentException ex)
        {
            // Compensate: release inventory, cancel order
            await CompensateAsync(ct);
            throw new SagaException("Payment failed", ex);
        }
    }
    
    private async Task CompensateAsync(CancellationToken ct)
    {
        State = SagaState.Compensating;
        
        // Compensate in reverse order
        if (_paymentId != 0)
            await _paymentService.RefundAsync(_paymentId, ct);
        
        if (_reservationId != 0)
            await _inventoryService.ReleaseReservationAsync(_reservationId, ct);
        
        if (_orderId != 0)
            await _orderService.CancelAsync(_orderId, "Saga compensation", ct);
        
        State = SagaState.Compensated;
    }
}
```
