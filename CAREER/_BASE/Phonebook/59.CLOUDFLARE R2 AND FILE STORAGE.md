## 59.01 — Cloudflare R2 with .NET

**What it is**
Cloudflare's S3-compatible object storage. Zero egress fees. Compatible with AWS S3 SDK.

**Setup**
```csharp
// NuGet: AWSSDK.S3

// Configuration in appsettings.json:
// {
//   "Cloudflare": {
//     "AccountId": "your-account-id",
//     "AccessKeyId": "your-access-key",
//     "SecretAccessKey": "stored-in-secrets-manager",
//     "BucketName": "my-bucket"
//   }
// }

builder.Services.AddSingleton<IAmazonS3>(sp =>
{
    var config = sp.GetRequiredService<IConfiguration>();
    var credentials = new BasicAWSCredentials(
        config["Cloudflare:AccessKeyId"],
        config["Cloudflare:SecretAccessKey"]);
    
    return new AmazonS3Client(credentials, new AmazonS3Config
    {
        ServiceURL = $"https://{config["Cloudflare:AccountId"]}.r2.cloudflarestorage.com",
        ForcePathStyle = true,  // Required for R2
        SignatureVersion = "4"
    });
});

builder.Services.AddScoped<IFileStorageService, R2FileStorageService>();
```

**File storage service**
```csharp
public class R2FileStorageService : IFileStorageService
{
    private readonly IAmazonS3 _s3;
    private readonly string _bucketName;
    private readonly ILogger<R2FileStorageService> _logger;
    
    public R2FileStorageService(IAmazonS3 s3, IConfiguration config, ILogger<R2FileStorageService> logger)
    {
        _s3 = s3;
        _bucketName = config["Cloudflare:BucketName"]!;
        _logger = logger;
    }
    
    public async Task<string> UploadAsync(
        Stream content,
        string fileName,
        string contentType,
        CancellationToken ct = default)
    {
        var key = $"{Guid.NewGuid():N}/{fileName}";
        
        var request = new PutObjectRequest
        {
            BucketName = _bucketName,
            Key = key,
            InputStream = content,
            ContentType = contentType,
            ServerSideEncryptionMethod = ServerSideEncryptionMethod.AES256,
            Metadata =
            {
                ["x-original-filename"] = fileName,
                ["x-uploaded-at"] = DateTime.UtcNow.ToString("O")
            }
        };
        
        await _s3.PutObjectAsync(request, ct);
        _logger.LogInformation("Uploaded file {Key} to R2", key);
        return key;
    }
    
    public async Task<Stream> DownloadAsync(string key, CancellationToken ct = default)
    {
        var response = await _s3.GetObjectAsync(_bucketName, key, ct);
        return response.ResponseStream;
    }
    
    public async Task DeleteAsync(string key, CancellationToken ct = default)
    {
        await _s3.DeleteObjectAsync(_bucketName, key, ct);
        _logger.LogInformation("Deleted file {Key} from R2", key);
    }
    
    public async Task<string> GeneratePresignedUrlAsync(
        string key,
        TimeSpan expiry,
        CancellationToken ct = default)
    {
        var request = new GetPreSignedUrlRequest
        {
            BucketName = _bucketName,
            Key = key,
            Expires = DateTime.UtcNow.Add(expiry),
            Protocol = Protocol.HTTPS
        };
        
        return _s3.GetPreSignedURL(request);
    }
    
    public async Task<string> GenerateUploadUrlAsync(
        string fileName,
        string contentType,
        long maxSizeBytes,
        TimeSpan expiry)
    {
        var key = $"uploads/{Guid.NewGuid():N}/{fileName}";
        
        // Presigned POST — allows direct upload from browser
        // Client uploads directly to R2 — doesn't go through your API server
        var conditions = new List<object>
        {
            new Dictionary<string, string> { ["bucket"] = _bucketName },
            new[] { "starts-with", "$key", "uploads/" },
            new[] { "content-length-range", "1", maxSizeBytes.ToString() },
            new Dictionary<string, string> { ["Content-Type"] = contentType }
        };
        
        // Return the presigned URL and key to client
        return GeneratePresignedUrlAsync(key, expiry).Result;
    }
    
    // Multipart upload for large files (> 100 MB)
    public async Task<string> UploadLargeFileAsync(
        Stream content,
        string fileName,
        string contentType,
        IProgress<double>? progress = null,
        CancellationToken ct = default)
    {
        var key = $"{Guid.NewGuid():N}/{fileName}";
        const int chunkSize = 5 * 1024 * 1024;  // 5 MB minimum part size
        
        // Initiate multipart upload
        var initiateResponse = await _s3.InitiateMultipartUploadAsync(new InitiateMultipartUploadRequest
        {
            BucketName = _bucketName,
            Key = key,
            ContentType = contentType
        }, ct);
        
        var uploadId = initiateResponse.UploadId;
        var parts = new List<PartETag>();
        long totalRead = 0;
        long contentLength = content.CanSeek ? content.Length : -1;
        
        try
        {
            var buffer = new byte[chunkSize];
            int partNumber = 1;
            
            while (true)
            {
                int bytesRead = await content.ReadAsync(buffer.AsMemory(), ct);
                if (bytesRead == 0) break;
                
                using var partStream = new MemoryStream(buffer, 0, bytesRead);
                var partResponse = await _s3.UploadPartAsync(new UploadPartRequest
                {
                    BucketName = _bucketName,
                    Key = key,
                    UploadId = uploadId,
                    PartNumber = partNumber++,
                    InputStream = partStream
                }, ct);
                
                parts.Add(new PartETag { PartNumber = partNumber - 1, ETag = partResponse.ETag });
                totalRead += bytesRead;
                
                if (contentLength > 0)
                    progress?.Report((double)totalRead / contentLength);
            }
            
            await _s3.CompleteMultipartUploadAsync(new CompleteMultipartUploadRequest
            {
                BucketName = _bucketName,
                Key = key,
                UploadId = uploadId,
                PartETags = parts
            }, ct);
            
            return key;
        }
        catch
        {
            await _s3.AbortMultipartUploadAsync(_bucketName, key, uploadId, ct);
            throw;
        }
    }
}
```

**Endpoint for file upload**
```csharp
app.MapPost("/api/files/upload", async (
    IFormFile file,
    IFileStorageService storage,
    CancellationToken ct) =>
{
    if (file.Length > 50 * 1024 * 1024)  // 50 MB limit
        return Results.BadRequest("File too large. Maximum size: 50 MB");
    
    var allowedTypes = new[] { "image/jpeg", "image/png", "application/pdf" };
    if (!allowedTypes.Contains(file.ContentType))
        return Results.BadRequest("File type not allowed");
    
    await using var stream = file.OpenReadStream();
    var key = await storage.UploadAsync(stream, file.FileName, file.ContentType, ct);
    
    return Results.Created($"/api/files/{Uri.EscapeDataString(key)}", new { Key = key });
})
.RequireAuthorization()
.DisableAntiforgery()  // Required for multipart form
.WithName("UploadFile");

// Get presigned download URL
app.MapGet("/api/files/{*key}", async (
    string key,
    IFileStorageService storage,
    CancellationToken ct) =>
{
    var url = await storage.GeneratePresignedUrlAsync(key, TimeSpan.FromHours(1), ct);
    return Results.Redirect(url);  // Or return the URL: Results.Ok(new { Url = url })
})
.RequireAuthorization();
```
