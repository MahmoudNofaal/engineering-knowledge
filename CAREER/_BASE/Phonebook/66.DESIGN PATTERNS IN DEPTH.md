## 66.01 — Creational Patterns

**Factory Method**
```csharp
// Define an interface for creating an object, but let subclasses decide which class to instantiate
public abstract class PaymentProcessorFactory
{
    public abstract IPaymentProcessor CreateProcessor();
    
    // Template method — uses the factory method
    public async Task<bool> ProcessPaymentAsync(decimal amount)
    {
        var processor = CreateProcessor();  // Factory method
        return await processor.ProcessAsync(amount);
    }
}

public class StripeProcessorFactory : PaymentProcessorFactory
{
    private readonly StripeConfig _config;
    public StripeProcessorFactory(StripeConfig config) => _config = config;
    
    public override IPaymentProcessor CreateProcessor()
        => new StripeProcessor(_config);
}

public class PayPalProcessorFactory : PaymentProcessorFactory
{
    public override IPaymentProcessor CreateProcessor()
        => new PayPalProcessor();
}

// In DI — the factory itself is injected
// Caller: new StripeProcessorFactory(config).ProcessPaymentAsync(amount)
```

**Abstract Factory**
```csharp
// Create families of related objects without specifying their concrete classes
public interface IUIFactory
{
    IButton CreateButton();
    ITextBox CreateTextBox();
    IDialog CreateDialog();
}

public class WindowsUIFactory : IUIFactory
{
    public IButton CreateButton() => new WindowsButton();
    public ITextBox CreateTextBox() => new WindowsTextBox();
    public IDialog CreateDialog() => new WindowsDialog();
}

public class MacOSUIFactory : IUIFactory
{
    public IButton CreateButton() => new MacButton();
    public ITextBox CreateTextBox() => new MacTextBox();
    public IDialog CreateDialog() => new MacDialog();
}

// Application uses the factory without knowing the concrete types
public class Application
{
    private readonly IUIFactory _factory;
    
    public Application(IUIFactory factory) => _factory = factory;
    
    public void BuildUI()
    {
        var button = _factory.CreateButton();  // Always Windows or Mac, never mixed
        var textBox = _factory.CreateTextBox();
        button.Render();
        textBox.Render();
    }
}
```

**Builder Pattern**
```csharp
// Construct complex objects step by step
public class HttpRequestBuilder
{
    private string _url = "";
    private HttpMethod _method = HttpMethod.Get;
    private readonly Dictionary<string, string> _headers = new();
    private string? _body;
    private TimeSpan _timeout = TimeSpan.FromSeconds(30);
    
    public HttpRequestBuilder WithUrl(string url) { _url = url; return this; }
    public HttpRequestBuilder WithMethod(HttpMethod method) { _method = method; return this; }
    public HttpRequestBuilder WithHeader(string key, string value) { _headers[key] = value; return this; }
    public HttpRequestBuilder WithBody(string body) { _body = body; return this; }
    public HttpRequestBuilder WithTimeout(TimeSpan timeout) { _timeout = timeout; return this; }
    public HttpRequestBuilder WithBearerToken(string token)
        => WithHeader("Authorization", $"Bearer {token}");
    public HttpRequestBuilder AsJson()
        => WithHeader("Content-Type", "application/json").WithHeader("Accept", "application/json");
    
    public HttpRequestMessage Build()
    {
        if (string.IsNullOrEmpty(_url)) throw new InvalidOperationException("URL is required");
        
        var request = new HttpRequestMessage(_method, _url);
        foreach (var (key, value) in _headers)
            request.Headers.TryAddWithoutValidation(key, value);
        if (_body is not null)
            request.Content = new StringContent(_body);
        
        return request;
    }
}

// Usage — fluent interface
var request = new HttpRequestBuilder()
    .WithUrl("https://api.example.com/orders")
    .WithMethod(HttpMethod.Post)
    .AsJson()
    .WithBearerToken(accessToken)
    .WithBody(JsonSerializer.Serialize(orderDto))
    .WithTimeout(TimeSpan.FromSeconds(10))
    .Build();
```

## 66.02 — Structural Patterns

**Adapter**
```csharp
// Convert an incompatible interface to a compatible one
// External library has this:
public class ExternalPaymentGateway
{
    public string MakePayment(string cardNumber, string cvv, double amount, string currencyCode)
        => Guid.NewGuid().ToString();  // Returns transaction ID
}

// Your system expects this:
public interface IPaymentProcessor
{
    Task<PaymentResult> ProcessAsync(PaymentRequest request, CancellationToken ct);
}

// Adapter bridges the gap
public class ExternalPaymentAdapter : IPaymentProcessor
{
    private readonly ExternalPaymentGateway _gateway;
    
    public ExternalPaymentAdapter(ExternalPaymentGateway gateway) => _gateway = gateway;
    
    public Task<PaymentResult> ProcessAsync(PaymentRequest request, CancellationToken ct)
    {
        // Adapt: convert PaymentRequest to the external format
        var transactionId = _gateway.MakePayment(
            request.CardNumber,
            request.CVV,
            (double)request.Amount,  // Adapt decimal to double
            request.Currency.Code);  // Adapt Money.Currency to string
        
        // Adapt: convert external result to our format
        return Task.FromResult(new PaymentResult(transactionId, true));
    }
}
```

**Composite**
```csharp
// Compose objects into tree structures
// Use case: file system (file and folder both implement same interface)
public interface IFileSystemItem
{
    string Name { get; }
    long GetSize();
    void Display(int depth = 0);
}

public class File : IFileSystemItem
{
    public string Name { get; }
    private readonly long _size;
    
    public File(string name, long size) { Name = name; _size = size; }
    public long GetSize() => _size;
    public void Display(int depth = 0) => Console.WriteLine($"{new string(' ', depth * 2)}{Name} ({_size:N0} bytes)");
}

public class Folder : IFileSystemItem
{
    public string Name { get; }
    private readonly List<IFileSystemItem> _children = new();
    
    public Folder(string name) { Name = name; }
    
    public void Add(IFileSystemItem item) => _children.Add(item);
    public void Remove(IFileSystemItem item) => _children.Remove(item);
    
    public long GetSize() => _children.Sum(c => c.GetSize());  // Recursive sum
    
    public void Display(int depth = 0)
    {
        Console.WriteLine($"{new string(' ', depth * 2)}[{Name}] ({GetSize():N0} bytes)");
        foreach (var child in _children)
            child.Display(depth + 1);
    }
}

// Usage
var root = new Folder("root");
root.Add(new File("README.md", 1024));
var src = new Folder("src");
src.Add(new File("Program.cs", 5120));
src.Add(new File("Startup.cs", 3072));
root.Add(src);

root.Display();
Console.WriteLine($"Total: {root.GetSize():N0} bytes");
```

**Flyweight**
```csharp
// Share common state among many objects to save memory
// Use case: game with thousands of trees — each tree has unique position but shared visual data

public class TreeType  // Shared (intrinsic) state
{
    public string Name { get; }
    public string Color { get; }
    public string Texture { get; }  // Expensive to store — shared!
    
    public TreeType(string name, string color, string texture)
    {
        Name = name; Color = color; Texture = texture;
    }
    
    public void Draw(int x, int y) { /* Draw tree at position using shared texture */ }
}

public class TreeFactory
{
    private static readonly Dictionary<string, TreeType> _types = new();
    
    public static TreeType GetTreeType(string name, string color, string texture)
    {
        var key = $"{name}_{color}_{texture}";
        if (!_types.TryGetValue(key, out var type))
        {
            type = new TreeType(name, color, texture);
            _types[key] = type;
        }
        return type;  // Return shared instance
    }
}

public class Tree  // Unique (extrinsic) state — position
{
    private readonly int _x, _y;
    private readonly TreeType _type;  // Reference to shared object
    
    public Tree(int x, int y, TreeType type) { _x = x; _y = y; _type = type; }
    
    public void Draw() => _type.Draw(_x, _y);
}

// 1000 oak trees: 1000 Tree objects (small) + 1 TreeType object (large texture) shared
```

## 66.03 — Behavioral Patterns

**Chain of Responsibility**
```csharp
// Pass a request through a chain of handlers
// Each handler decides to process or pass to next
public abstract class RequestHandler<TRequest>
{
    private RequestHandler<TRequest>? _next;
    
    public RequestHandler<TRequest> SetNext(RequestHandler<TRequest> next)
    {
        _next = next;
        return next;
    }
    
    public virtual Task HandleAsync(TRequest request, CancellationToken ct)
        => _next?.HandleAsync(request, ct) ?? Task.CompletedTask;
}

public class AuthenticationHandler : RequestHandler<HttpRequest>
{
    public override async Task HandleAsync(HttpRequest request, CancellationToken ct)
    {
        if (!request.Headers.ContainsKey("Authorization"))
            throw new UnauthorizedException();
        
        await base.HandleAsync(request, ct);  // Pass to next
    }
}

public class RateLimitHandler : RequestHandler<HttpRequest>
{
    private readonly IRateLimiter _limiter;
    
    public override async Task HandleAsync(HttpRequest request, CancellationToken ct)
    {
        var clientId = ExtractClientId(request);
        if (!await _limiter.IsAllowedAsync(clientId, ct))
            throw new RateLimitExceededException();
        
        await base.HandleAsync(request, ct);
    }
}

// Build chain
var chain = new AuthenticationHandler();
chain.SetNext(new RateLimitHandler())
     .SetNext(new BusinessLogicHandler());

await chain.HandleAsync(httpRequest, ct);
```

**Command Pattern**
```csharp
// Encapsulate operations as objects — enables undo/redo, queuing, logging
public interface ICommand
{
    void Execute();
    void Undo();
}

public class MoveCommand : ICommand
{
    private readonly Document _doc;
    private readonly int _dx, _dy;
    
    public MoveCommand(Document doc, int dx, int dy)
    {
        _doc = doc; _dx = dx; _dy = dy;
    }
    
    public void Execute() { _doc.X += _dx; _doc.Y += _dy; }
    public void Undo() { _doc.X -= _dx; _doc.Y -= _dy; }
}

public class CommandHistory
{
    private readonly Stack<ICommand> _history = new();
    
    public void Execute(ICommand command)
    {
        command.Execute();
        _history.Push(command);
    }
    
    public void Undo()
    {
        if (_history.TryPop(out var command))
            command.Undo();
    }
}
```

**Iterator Pattern (built into C#)**
```csharp
// C# IEnumerable<T> / yield return IS the iterator pattern

// Custom iterator
public class NumberRange : IEnumerable<int>
{
    private readonly int _start, _end, _step;
    
    public NumberRange(int start, int end, int step = 1)
    {
        _start = start; _end = end; _step = step;
    }
    
    public IEnumerator<int> GetEnumerator()
    {
        for (int i = _start; i <= _end; i += _step)
            yield return i;
    }
    
    IEnumerator IEnumerable.GetEnumerator() => GetEnumerator();
}

// Usage
foreach (var n in new NumberRange(0, 100, 5))
    Console.Write(n + " ");  // 0 5 10 15 ... 100

// Infinite iterator
public static IEnumerable<int> FibonacciSequence()
{
    int a = 0, b = 1;
    while (true)
    {
        yield return a;
        (a, b) = (b, a + b);
    }
}

var first10Fibs = FibonacciSequence().Take(10).ToList();
```

**Visitor Pattern**
```csharp
// Add operations to object hierarchies without modifying the classes
public interface IShapeVisitor
{
    double VisitCircle(Circle circle);
    double VisitRectangle(Rectangle rectangle);
    double VisitTriangle(Triangle triangle);
}

public abstract class Shape
{
    public abstract double Accept(IShapeVisitor visitor);
}

public class Circle : Shape
{
    public double Radius { get; }
    public Circle(double radius) => Radius = radius;
    public override double Accept(IShapeVisitor visitor) => visitor.VisitCircle(this);
}

public class Rectangle : Shape
{
    public double Width { get; }
    public double Height { get; }
    public Rectangle(double width, double height) { Width = width; Height = height; }
    public override double Accept(IShapeVisitor visitor) => visitor.VisitRectangle(this);
}

// New operation — no shape classes modified
public class AreaCalculator : IShapeVisitor
{
    public double VisitCircle(Circle c) => Math.PI * c.Radius * c.Radius;
    public double VisitRectangle(Rectangle r) => r.Width * r.Height;
    public double VisitTriangle(Triangle t) => 0.5 * t.Base * t.Height;
}

// Another new operation
public class PerimeterCalculator : IShapeVisitor
{
    public double VisitCircle(Circle c) => 2 * Math.PI * c.Radius;
    public double VisitRectangle(Rectangle r) => 2 * (r.Width + r.Height);
    public double VisitTriangle(Triangle t) => t.A + t.B + t.C;
}

// Usage
var shapes = new Shape[] { new Circle(5), new Rectangle(3, 4) };
var areaCalc = new AreaCalculator();
var totalArea = shapes.Sum(s => s.Accept(areaCalc));
```
