## 14.01 — DbContext and Change Tracking

**DbContext lifecycle**
In ASP.NET Core, DbContext is registered as Scoped — one instance per HTTP request.

```csharp
// Registration
builder.Services.AddDbContext<AppDbContext>(options =>
    options.UseSqlServer(
        builder.Configuration.GetConnectionString("Default"),
        sql => sql.EnableRetryOnFailure(3)));

// Configuration
public class AppDbContext : DbContext
{
    public DbSet<Order> Orders { get; set; } = null!;
    public DbSet<Product> Products { get; set; } = null!;
    
    public AppDbContext(DbContextOptions<AppDbContext> options) : base(options) { }
    
    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        modelBuilder.ApplyConfigurationsFromAssembly(Assembly.GetExecutingAssembly());
    }
}

// Entity configuration (Fluent API — preferred over DataAnnotations)
public class OrderConfiguration : IEntityTypeConfiguration<Order>
{
    public void Configure(EntityTypeBuilder<Order> builder)
    {
        builder.ToTable("Orders");
        builder.HasKey(o => o.Id);
        builder.Property(o => o.Status)
            .HasConversion<string>()
            .HasMaxLength(50);
        builder.Property(o => o.Total)
            .HasColumnType("decimal(18, 2)");
        builder.HasMany(o => o.Items)
            .WithOne(i => i.Order)
            .HasForeignKey(i => i.OrderId)
            .OnDelete(DeleteBehavior.Cascade);
        builder.HasIndex(o => o.CustomerId);
    }
}
```

**Change tracking states**
- `Added`: New entity, will INSERT on SaveChanges
- `Modified`: Existing entity with changes, will UPDATE
- `Deleted`: Will DELETE on SaveChanges
- `Unchanged`: No changes
- `Detached`: Not tracked by the context

```csharp
var entry = _db.Entry(order);
Console.WriteLine(entry.State);  // EntityState.Modified, etc.

// Explicitly set state
_db.Entry(order).State = EntityState.Modified;

// Detach to stop tracking
_db.Entry(order).State = EntityState.Detached;
```

**AsNoTracking**
```csharp
// Read-only query — no change tracking overhead
var orders = await _db.Orders
    .AsNoTracking()
    .Where(o => o.CustomerId == customerId)
    .ToListAsync();

// Global no-tracking query
options.UseQueryTrackingBehavior(QueryTrackingBehavior.NoTracking);
```

## 14.02 — The N+1 Problem

**What it is**
When loading a collection of entities, EF Core makes one query for the list, then one query for each entity's related data. N items = N+1 queries.

**The problem**
```csharp
// This generates N+1 queries!
var orders = await _db.Orders.ToListAsync();  // Query 1: SELECT * FROM Orders

foreach (var order in orders)
{
    // Query per order: SELECT * FROM OrderItems WHERE OrderId = @id
    var items = order.Items;  // Lazy loading triggers here
}

// 100 orders = 101 queries
```

**Solution 1: Eager loading**
```csharp
var orders = await _db.Orders
    .Include(o => o.Items)
        .ThenInclude(i => i.Product)
    .Include(o => o.Customer)
    .Where(o => o.Status == OrderStatus.Pending)
    .ToListAsync();
// 1 query with JOINs
```

**Solution 2: AsSplitQuery (avoids cartesian explosion)**
```csharp
// Single query with multiple includes can cause cartesian product explosion
// AsSplitQuery sends separate queries but avoids N+1
var orders = await _db.Orders
    .Include(o => o.Items)
    .Include(o => o.Tags)
    .AsSplitQuery()  // Two queries: orders + items, orders + tags
    .ToListAsync();
```

**Solution 3: Projection (best for read models)**
```csharp
var orderDtos = await _db.Orders
    .Where(o => o.CustomerId == customerId)
    .Select(o => new OrderSummaryDto
    {
        OrderId = o.Id,
        Total = o.Items.Sum(i => i.Price * i.Quantity),
        ItemCount = o.Items.Count,
        Status = o.Status.ToString()
    })
    .ToListAsync();
// EF Core translates the Select into optimized SQL - no extra queries
```

**Solution 4: Explicit loading**
```csharp
var order = await _db.Orders.FindAsync(orderId);
await _db.Entry(order).Collection(o => o.Items).LoadAsync();
await _db.Entry(order).Reference(o => o.Customer).LoadAsync();
```

**Detecting N+1**
```csharp
// Log all SQL queries in development
optionsBuilder.LogTo(Console.WriteLine, LogLevel.Information)
    .EnableSensitiveDataLogging();

// Or use EF Core interceptors to count queries per request
```

## 14.03 — Raw SQL and Bulk Operations

**When to use raw SQL**
- Complex queries EF Core can't translate
- Performance-critical paths where ORM overhead is measurable
- Bulk operations

```csharp
// Query with raw SQL (parameterized - safe from injection)
var orders = await _db.Orders
    .FromSqlRaw("SELECT * FROM Orders WHERE CustomerId = {0} AND Status = {1}",
        customerId, "Pending")
    .Include(o => o.Items)
    .ToListAsync();

// Interpolated SQL (SAFE - params are parameterized)
var orders = await _db.Orders
    .FromSqlInterpolated($"SELECT * FROM Orders WHERE CustomerId = {customerId}")
    .ToListAsync();

// DANGEROUS - SQL injection possible:
var orders = await _db.Orders
    .FromSqlRaw($"SELECT * FROM Orders WHERE CustomerId = {customerId}")
    // ^ String interpolation before parameterization!
    .ToListAsync();

// Non-query SQL
int affected = await _db.Database.ExecuteSqlRawAsync(
    "UPDATE Orders SET Status = 'Archived' WHERE OrderDate < {0}",
    DateTime.UtcNow.AddYears(-2));

// EF Core 7+ bulk operations (no round-trips per entity)
await _db.Orders
    .Where(o => o.Status == OrderStatus.Cancelled)
    .ExecuteDeleteAsync();

await _db.Orders
    .Where(o => o.Status == OrderStatus.Pending && o.OrderDate < deadline)
    .ExecuteUpdateAsync(s => s
        .SetProperty(o => o.Status, OrderStatus.Cancelled)
        .SetProperty(o => o.CancelledAt, DateTime.UtcNow));
```

## 14.04 — EF Core Interceptors

**What it is**
Hooks into EF Core's pipeline for cross-cutting concerns like audit trails, soft deletes, and performance monitoring.

```csharp
// Audit trail interceptor
public class AuditInterceptor : SaveChangesInterceptor
{
    private readonly ICurrentUserService _currentUser;
    
    public override InterceptionResult<int> SavingChanges(
        DbContextEventData eventData, InterceptionResult<int> result)
    {
        if (eventData.Context is null) return result;
        
        foreach (var entry in eventData.Context.ChangeTracker.Entries())
        {
            if (entry.Entity is IAuditable auditable)
            {
                if (entry.State == EntityState.Added)
                {
                    auditable.CreatedAt = DateTime.UtcNow;
                    auditable.CreatedBy = _currentUser.UserId;
                }
                if (entry.State == EntityState.Modified)
                {
                    auditable.ModifiedAt = DateTime.UtcNow;
                    auditable.ModifiedBy = _currentUser.UserId;
                }
            }
        }
        return result;
    }
}

// Register
builder.Services.AddDbContext<AppDbContext>((sp, options) =>
{
    options.UseSqlServer(connectionString)
           .AddInterceptors(sp.GetRequiredService<AuditInterceptor>());
});
```

## 14.05 — Global Query Filters

**What it is**
Filters applied to all queries for a given entity type. Used for soft delete and multi-tenancy.

```csharp
// Soft delete filter
protected override void OnModelCreating(ModelBuilder modelBuilder)
{
    modelBuilder.Entity<Order>()
        .HasQueryFilter(o => !o.IsDeleted);  // Automatically applied to all Order queries
    
    // Multi-tenancy
    modelBuilder.Entity<Product>()
        .HasQueryFilter(p => p.TenantId == _tenantId);  // _tenantId from service
}

// Override filter for specific query
var allOrders = await _db.Orders
    .IgnoreQueryFilters()  // Ignore soft delete filter
    .ToListAsync();
```
