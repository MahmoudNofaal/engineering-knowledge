## 34.01 — Architecture Tests with NetArchTest

**What it is**
Code-as-specification tests that enforce architectural rules — like "no class in Application layer references Infrastructure".

```csharp
// NuGet: NetArchTest.Rules

public class ArchitectureTests
{
    private const string DomainAssembly = "MyApp.Domain";
    private const string ApplicationAssembly = "MyApp.Application";
    private const string InfrastructureAssembly = "MyApp.Infrastructure";
    private const string ApiAssembly = "MyApp.Api";
    
    [Fact]
    public void Domain_Should_Not_Reference_Application()
    {
        var result = Types.InAssembly(Assembly.Load(DomainAssembly))
            .Should()
            .NotHaveDependencyOn(ApplicationAssembly)
            .GetResult();
        
        result.IsSuccessful.Should().BeTrue(
            because: $"Domain should not reference Application: {string.Join(", ", result.FailingTypes?.Select(t => t.Name) ?? Array.Empty<string>())}");
    }
    
    [Fact]
    public void Application_Should_Not_Reference_Infrastructure()
    {
        var result = Types.InAssembly(Assembly.Load(ApplicationAssembly))
            .Should()
            .NotHaveDependencyOn(InfrastructureAssembly)
            .GetResult();
        
        result.IsSuccessful.Should().BeTrue();
    }
    
    [Fact]
    public void Domain_Entities_Should_Be_In_Entities_Namespace()
    {
        var result = Types.InAssembly(Assembly.Load(DomainAssembly))
            .That().HaveNameEndingWith("Entity")
            .Should()
            .ResideInNamespace($"{DomainAssembly}.Entities")
            .GetResult();
        
        result.IsSuccessful.Should().BeTrue();
    }
    
    [Fact]
    public void All_Handlers_Should_Be_Sealed()
    {
        var result = Types.InAssembly(Assembly.Load(ApplicationAssembly))
            .That().ImplementInterface(typeof(IRequestHandler<,>))
            .Should()
            .BeSealed()
            .GetResult();
        
        result.IsSuccessful.Should().BeTrue();
    }
    
    [Fact]
    public void Services_Should_Have_Matching_Interfaces()
    {
        var result = Types.InAssembly(Assembly.Load(ApplicationAssembly))
            .That().HaveNameEndingWith("Service")
            .Should()
            .HaveNameStartingWith("I")
            .Or()
            .MeetCustomRule(new HasMatchingInterfaceRule())
            .GetResult();
    }
}
```

## 34.02 — Mutation Testing

**What it is**
A technique for measuring test suite quality. A tool introduces small "mutations" (bugs) into your code. If your tests catch the mutation, the mutant is "killed". If not, the mutant "survives" — a gap in test coverage.

**Stryker.NET**
```bash
dotnet tool install -g dotnet-stryker
dotnet stryker --project MyApp.Application
```

**What mutations Stryker introduces**
- `>` → `>=` and `<` (boundary mutations)
- `+` → `-`, `*`, `/` (arithmetic mutations)
- `&&` → `||` (logical mutations)
- `true` → `false` (boolean literal mutations)
- Remove method calls
- Change return values

**Example**
```csharp
// Original
public bool IsEligibleForDiscount(Order order)
    => order.Total > 100 && order.Customer.IsVIP;

// Mutation 1: order.Total >= 100 (survived = tests don't check boundary)
// Mutation 2: order.Total > 100 || order.Customer.IsVIP (survived = tests don't check AND)
// Mutation 3: order.Total <= 100 (killed = tests catch wrong comparison)
```

**High mutation score ≠ good tests**
Mutation testing reveals whether your tests actually assert the right things, not just whether they execute the code.

## 34.03 — Property-Based Testing

**What it is**
Instead of writing specific test cases, you write properties (invariants) that should hold for all inputs. The framework generates random inputs and finds counterexamples.

```csharp
// NuGet: FsCheck.NUnit or Hedgehog.CSharp or CsCheck

// Property: sorting a list produces a sorted list
[Property]
public bool Sorting_ProducesSortedList(int[] arr)
{
    var sorted = arr.OrderBy(x => x).ToArray();
    for (int i = 0; i < sorted.Length - 1; i++)
        if (sorted[i] > sorted[i + 1]) return false;
    return true;
}

// Property: adding then removing an item from a set is idempotent
[Property]
public bool HashSet_AddThenRemove_IsIdempotent(int[] initial, int element)
{
    var set = new HashSet<int>(initial);
    set.Add(element);
    set.Remove(element);
    
    // Element should not be in set if it wasn't originally
    var wasOriginallyPresent = initial.Contains(element);
    return set.Contains(element) == wasOriginallyPresent;
}

// Property for domain model
[Property]
public bool Order_Total_EqualsSum_OfItems(NonEmptyArray<(PositiveInt price, PositiveInt qty)> items)
{
    var order = Order.Create(1);
    foreach (var (price, qty) in items.Get)
        order.AddItem(1, price.Get, qty.Get);
    
    var expectedTotal = items.Get.Sum(i => (decimal)i.price.Get * i.qty.Get);
    return order.Total == expectedTotal;
}
```
