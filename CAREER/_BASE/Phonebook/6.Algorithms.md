## 6.01 — Big-O Notation

| Notation | Name | Example |
|---|---|---|
| O(1) | Constant | Array access, hash map lookup |
| O(log n) | Logarithmic | Binary search |
| O(n) | Linear | Linear scan |
| O(n log n) | Linearithmic | Merge sort |
| O(n²) | Quadratic | Nested loops |
| O(2ⁿ) | Exponential | Brute-force subsets |
| O(n!) | Factorial | Brute-force permutations |

**At n=1,000,000**: O(log n) = 20 ops. O(n²) = 10^12 ops — impossible.

**Amortized**: List<T>.Add is O(1) amortized (O(n) worst case during resize, but rare).

## 6.02 — Binary Search

```csharp
// Standard
int BinarySearch(int[] nums, int target)
{
    int left = 0, right = nums.Length - 1;
    while (left <= right)
    {
        int mid = left + (right - left) / 2;
        if (nums[mid] == target) return mid;
        if (nums[mid] < target) left = mid + 1;
        else right = mid - 1;
    }
    return -1;
}

// Binary search on the answer
int ShipWithinDays(int[] weights, int days)
{
    int left = weights.Max();
    int right = weights.Sum();
    
    while (left < right)
    {
        int mid = left + (right - left) / 2;
        if (CanShip(weights, days, mid)) right = mid;
        else left = mid + 1;
    }
    return left;
}

bool CanShip(int[] weights, int days, int capacity)
{
    int needed = 1, current = 0;
    foreach (int w in weights)
    {
        if (current + w > capacity) { needed++; current = 0; }
        current += w;
    }
    return needed <= days;
}

// Rotated sorted array
int SearchRotated(int[] nums, int target)
{
    int left = 0, right = nums.Length - 1;
    while (left <= right)
    {
        int mid = left + (right - left) / 2;
        if (nums[mid] == target) return mid;
        
        if (nums[left] <= nums[mid])
        {
            if (nums[left] <= target && target < nums[mid]) right = mid - 1;
            else left = mid + 1;
        }
        else
        {
            if (nums[mid] < target && target <= nums[right]) left = mid + 1;
            else right = mid - 1;
        }
    }
    return -1;
}
```

## 6.03 — Two Pointers

```csharp
// Opposite ends — two sum sorted array
bool TwoSum(int[] nums, int target)
{
    int left = 0, right = nums.Length - 1;
    while (left < right)
    {
        int sum = nums[left] + nums[right];
        if (sum == target) return true;
        if (sum < target) left++;
        else right--;
    }
    return false;
}

// Same direction — remove duplicates in-place
int RemoveDuplicates(int[] nums)
{
    if (nums.Length == 0) return 0;
    int slow = 0;
    for (int fast = 1; fast < nums.Length; fast++)
        if (nums[fast] != nums[slow])
            nums[++slow] = nums[fast];
    return slow + 1;
}

// Fast/slow — detect cycle
bool HasCycle(ListNode head)
{
    var slow = head;
    var fast = head;
    while (fast?.Next != null)
    {
        slow = slow.Next;
        fast = fast.Next.Next;
        if (slow == fast) return true;
    }
    return false;
}
```

## 6.04 — Sliding Window

```csharp
// Fixed size — max sum subarray of size k
int MaxSumSubarray(int[] nums, int k)
{
    int sum = nums.Take(k).Sum();
    int max = sum;
    for (int i = k; i < nums.Length; i++)
    {
        sum += nums[i] - nums[i - k];
        max = Math.Max(max, sum);
    }
    return max;
}

// Variable size — longest substring without repeating
int LongestSubstring(string s)
{
    var seen = new Dictionary<char, int>();
    int left = 0, max = 0;
    for (int right = 0; right < s.Length; right++)
    {
        if (seen.TryGetValue(s[right], out int idx) && idx >= left)
            left = idx + 1;
        seen[s[right]] = right;
        max = Math.Max(max, right - left + 1);
    }
    return max;
}
```

## 6.05 — Dynamic Programming

**Recognize DP**: overlapping subproblems + optimal substructure. Key words: minimum/maximum/count of ways.

```csharp
// Coin change — minimum coins
int CoinChange(int[] coins, int amount)
{
    var dp = Enumerable.Repeat(amount + 1, amount + 1).ToArray();
    dp[0] = 0;
    for (int i = 1; i <= amount; i++)
        foreach (int coin in coins)
            if (coin <= i)
                dp[i] = Math.Min(dp[i], dp[i - coin] + 1);
    return dp[amount] > amount ? -1 : dp[amount];
}

// LCS — longest common subsequence
int LCS(string s1, string s2)
{
    int m = s1.Length, n = s2.Length;
    var dp = new int[m + 1, n + 1];
    for (int i = 1; i <= m; i++)
        for (int j = 1; j <= n; j++)
            dp[i, j] = s1[i-1] == s2[j-1]
                ? dp[i-1, j-1] + 1
                : Math.Max(dp[i-1, j], dp[i, j-1]);
    return dp[m, n];
}

// 0/1 Knapsack
int Knapsack(int[] weights, int[] values, int capacity)
{
    int n = weights.Length;
    var dp = new int[n + 1, capacity + 1];
    for (int i = 1; i <= n; i++)
        for (int w = 0; w <= capacity; w++)
        {
            dp[i, w] = dp[i-1, w];
            if (weights[i-1] <= w)
                dp[i, w] = Math.Max(dp[i, w], dp[i-1, w - weights[i-1]] + values[i-1]);
        }
    return dp[n, capacity];
}
```

## 6.06 — Backtracking

```csharp
// Template
void Backtrack(State current, List<State> result)
{
    if (IsComplete(current)) { result.Add(Copy(current)); return; }
    foreach (var choice in GetChoices(current))
    {
        MakeChoice(choice, current);
        Backtrack(current, result);
        UndoChoice(choice, current);
    }
}

// Permutations
IList<IList<int>> Permute(int[] nums)
{
    var result = new List<IList<int>>();
    var used = new bool[nums.Length];
    
    void BT(List<int> curr)
    {
        if (curr.Count == nums.Length) { result.Add(new List<int>(curr)); return; }
        for (int i = 0; i < nums.Length; i++)
        {
            if (used[i]) continue;
            used[i] = true;
            curr.Add(nums[i]);
            BT(curr);
            curr.RemoveAt(curr.Count - 1);
            used[i] = false;
        }
    }
    
    BT(new List<int>());
    return result;
}
```
