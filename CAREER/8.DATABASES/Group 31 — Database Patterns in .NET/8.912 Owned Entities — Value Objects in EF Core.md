# Owned Entities — Value Objects in EF Core

## 1 — Overview

Owned entity types (also called "value objects" in Domain-Driven Design) are types that do not have their own identity. They are owned by another entity and are stored as part of that entity's table row (table splitting) or in a separate table. In EF Core, owned types are configured via OwnsOne and OwnsMany.

Common examples: Address, Money, OrderLine, PersonName, EmailAddress, Period.

| Feature | EF Core OwnsOne / OwnsMany | Dapper |
|---------|---------------------------|--------|
| Identity | No identity — lifecycle tied to owner | No identity — manual mapping |
| Table mapping | Same table (columns) or separate table | Same columns or separate table |
| Change tracking | Automatic | N/A (no change tracker) |
| Querying | .Include() or auto-included | Manual JOIN + mapping |
| Value equality | Override Equals / GetHashCode | Manual equality |
| Nested owned types | Supported (deep nesting) | Manual mapping |

## 2 — Configuration — OwnsOne (Single Value Object)

### 2.1 — Basic OwnsOne Configuration

`csharp
public class Customer
{
    public int CustomerId { get; set; }
    public string Name { get; set; }
    public EmailAddress Email { get; set; }
    public Address ShippingAddress { get; set; }
    public Address BillingAddress { get; set; }
    public Money AccountBalance { get; set; }
}

public class Address
{
    public string Street { get; set; }
    public string City { get; set; }
    public string State { get; set; }
    public string ZipCode { get; set; }
    public string Country { get; set; }
}

public class EmailAddress
{
    public string Value { get; set; }
    private EmailAddress() { }
    public EmailAddress(string value)
    {
        if (string.IsNullOrWhiteSpace(value))
            throw new ArgumentException("Email cannot be empty", nameof(value));
        Value = value;
    }
}

public class Money
{
    public decimal Amount { get; set; }
    public string Currency { get; set; }
}

public class AppDbContext : DbContext
{
    public DbSet<Customer> Customers { get; set; }

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        modelBuilder.Entity<Customer>(entity =>
        {
            entity.HasKey(c => c.CustomerId);
            entity.Property(c => c.Name).HasMaxLength(200).IsRequired();

            entity.OwnsOne(c => c.ShippingAddress, address =>
            {
                address.Property(a => a.Street).HasColumnName("ShippingStreet").HasMaxLength(200);
                address.Property(a => a.City).HasColumnName("ShippingCity").HasMaxLength(100);
                address.Property(a => a.State).HasColumnName("ShippingState").HasMaxLength(50);
                address.Property(a => a.ZipCode).HasColumnName("ShippingZipCode").HasMaxLength(20);
                address.Property(a => a.Country).HasColumnName("ShippingCountry").HasMaxLength(100);
            });

            entity.OwnsOne(c => c.BillingAddress, address =>
            {
                address.Property(a => a.Street).HasColumnName("BillingStreet").HasMaxLength(200);
                address.Property(a => a.City).HasColumnName("BillingCity").HasMaxLength(100);
                address.Property(a => a.State).HasColumnName("BillingState").HasMaxLength(50);
                address.Property(a => a.ZipCode).HasColumnName("BillingZipCode").HasMaxLength(20);
                address.Property(a => a.Country).HasColumnName("BillingCountry").HasMaxLength(100);
            });

            entity.OwnsOne(c => c.Email, email =>
            {
                email.Property(e => e.Value).HasColumnName("Email").HasMaxLength(200).IsRequired();
            });

            entity.OwnsOne(c => c.AccountBalance, money =>
            {
                money.Property(m => m.Amount).HasColumnName("BalanceAmount").HasColumnType("decimal(18,2)");
                money.Property(m => m.Currency).HasColumnName("BalanceCurrency").HasMaxLength(3);
            });
        });
    }
}
`

## 3 — Configuration — OwnsMany (Collection)

### 3.1 — OwnsMany Configuration

`csharp
public class Order
{
    public int OrderId { get; set; }
    public DateTime OrderDate { get; set; }
    public decimal TotalAmount { get; set; }
    public List<OrderLine> OrderLines { get; set; } = new();
    public ShippingInfo ShippingInfo { get; set; }
}

public class OrderLine
{
    public string ProductName { get; set; }
    public int Quantity { get; set; }
    public decimal UnitPrice { get; set; }
}

public class ShippingInfo
{
    public string Carrier { get; set; }
    public string TrackingNumber { get; set; }
    public decimal Cost { get; set; }
}

protected override void OnModelCreating(ModelBuilder modelBuilder)
{
    modelBuilder.Entity<Order>(entity =>
    {
        entity.HasKey(o => o.OrderId);
        entity.Property(o => o.OrderDate).IsRequired();
        entity.Property(o => o.TotalAmount).HasColumnType("decimal(18,2)");

        entity.OwnsMany(o => o.OrderLines, line =>
        {
            line.WithOwner().HasForeignKey("OrderId");
            line.Property(l => l.ProductName).HasMaxLength(200).IsRequired();
            line.Property(l => l.Quantity).IsRequired();
            line.Property(l => l.UnitPrice).HasColumnType("decimal(18,2)");
            line.ToTable("OrderLines");
        });

        entity.OwnsOne(o => o.ShippingInfo, shipping =>
        {
            shipping.Property(s => s.Carrier).HasColumnName("ShippingCarrier").HasMaxLength(100);
            shipping.Property(s => s.TrackingNumber).HasColumnName("TrackingNumber").HasMaxLength(100);
            shipping.Property(s => s.Cost).HasColumnName("ShippingCost").HasColumnType("decimal(18,2)");
        });
    });
}
`

### 3.2 — Generated SQL for OwnsMany

`sql
CREATE TABLE Orders (
    OrderId INT IDENTITY(1,1) PRIMARY KEY,
    OrderDate DATETIME2 NOT NULL,
    TotalAmount DECIMAL(18,2) NOT NULL,
    ShippingCarrier NVARCHAR(100) NULL,
    TrackingNumber NVARCHAR(100) NULL,
    ShippingCost DECIMAL(18,2) NULL
);

CREATE TABLE OrderLines (
    OrderId INT NOT NULL,
    Id INT IDENTITY(1,1) NOT NULL,
    ProductName NVARCHAR(200) NOT NULL,
    Quantity INT NOT NULL,
    UnitPrice DECIMAL(18,2) NOT NULL,
    CONSTRAINT PK_OrderLines PRIMARY KEY (OrderId, Id),
    CONSTRAINT FK_OrderLines_Orders FOREIGN KEY (OrderId) REFERENCES Orders(OrderId)
);
`

### 4.3 — Dapper — Flattened Mapping

`csharp
public class CustomerFlatDto
{
    public int CustomerId { get; set; }
    public string Name { get; set; }
    public string ShippingStreet { get; set; }
    public string ShippingCity { get; set; }
    public string ShippingState { get; set; }
    public string ShippingZipCode { get; set; }
    public string ShippingCountry { get; set; }
    public string BillingStreet { get; set; }
    public string BillingCity { get; set; }
    public string BillingState { get; set; }
    public string BillingZipCode { get; set; }
    public string BillingCountry { get; set; }
    public string Email { get; set; }
    public decimal BalanceAmount { get; set; }
    public string BalanceCurrency { get; set; }
}

public class CustomerRepository
{
    private readonly string _connectionString;

    public CustomerRepository(string connectionString)
    {
        _connectionString = connectionString;
    }

    public async Task<Customer> GetCustomerAsync(int customerId)
    {
        const string sql = @"
            SELECT
                CustomerId, Name,
                ShippingStreet, ShippingCity, ShippingState, ShippingZipCode, ShippingCountry,
                BillingStreet, BillingCity, BillingState, BillingZipCode, BillingCountry,
                Email,
                BalanceAmount, BalanceCurrency
            FROM Customers
            WHERE CustomerId = @CustomerId";

        await using var conn = new SqlConnection(_connectionString);
        var dto = await conn.QuerySingleOrDefaultAsync<CustomerFlatDto>(sql, new { CustomerId = customerId });
        if (dto == null) return null;

        return new Customer
        {
            CustomerId = dto.CustomerId,
            Name = dto.Name,
            Email = new EmailAddress(dto.Email),
            ShippingAddress = new Address
            {
                Street = dto.ShippingStreet,
                City = dto.ShippingCity,
                State = dto.ShippingState,
                ZipCode = dto.ShippingZipCode,
                Country = dto.ShippingCountry
            },
            BillingAddress = new Address
            {
                Street = dto.BillingStreet,
                City = dto.BillingCity,
                State = dto.BillingState,
                ZipCode = dto.BillingZipCode,
                Country = dto.BillingCountry
            },
            AccountBalance = new Money
            {
                Amount = dto.BalanceAmount,
                Currency = dto.BalanceCurrency
            }
        };
    }

    public async Task<int> InsertCustomerAsync(Customer customer)
    {
        const string sql = @"
            INSERT INTO Customers (
                Name,
                ShippingStreet, ShippingCity, ShippingState, ShippingZipCode, ShippingCountry,
                BillingStreet, BillingCity, BillingState, BillingZipCode, BillingCountry,
                Email,
                BalanceAmount, BalanceCurrency
            ) VALUES (
                @Name,
                @ShippingStreet, @ShippingCity, @ShippingState, @ShippingZipCode, @ShippingCountry,
                @BillingStreet, @BillingCity, @BillingState, @BillingZipCode, @BillingCountry,
                @Email,
                @BalanceAmount, @BalanceCurrency
            );
            SELECT CAST(SCOPE_IDENTITY() AS INT);";

        var parameters = new
        {
            customer.Name,
            customer.ShippingAddress?.Street,
            customer.ShippingAddress?.City,
            customer.ShippingAddress?.State,
            customer.ShippingAddress?.ZipCode,
            customer.ShippingAddress?.Country,
            customer.BillingAddress?.Street,
            customer.BillingAddress?.City,
            customer.BillingAddress?.State,
            customer.BillingAddress?.ZipCode,
            customer.BillingAddress?.Country,
            Email = customer.Email?.Value,
            customer.AccountBalance?.Amount,
            customer.AccountBalance?.Currency
        };

        await using var conn = new SqlConnection(_connectionString);
        return await conn.QuerySingleAsync<int>(sql, parameters);
    }
}
`

### 4.4 — Dapper Multi-Mapping for OwnsMany

`csharp
public async Task<Order> GetOrderWithLinesAsync(int orderId)
{
    const string sql = @"
        SELECT
            o.OrderId, o.OrderDate, o.TotalAmount,
            o.ShippingCarrier, o.TrackingNumber, o.ShippingCost,
            ol.Id AS LineId, ol.ProductName, ol.Quantity, ol.UnitPrice
        FROM Orders o
        LEFT JOIN OrderLines ol ON o.OrderId = ol.OrderId
        WHERE o.OrderId = @OrderId
        ORDER BY ol.Id";

    await using var conn = new SqlConnection(_connectionString);

    var orderDictionary = new Dictionary<int, Order>();

    var result = await conn.QueryAsync<Order, OrderLine, Order>(
        sql,
        (order, line) =>
        {
            if (!orderDictionary.TryGetValue(order.OrderId, out var existingOrder))
            {
                existingOrder = order;
                existingOrder.OrderLines = new List<OrderLine>();
                orderDictionary.Add(existingOrder.OrderId, existingOrder);
            }
            if (line != null)
            {
                existingOrder.OrderLines.Add(line);
            }
            return existingOrder;
        },
        new { OrderId = orderId },
        splitOn: "LineId");

    return orderDictionary.Values.FirstOrDefault();
}
`

## 6 — Gotchas

### 6.1 — Owned Entities Cannot Be Tracked Independently

`csharp
var line = await context.Set<OrderLine>().FirstAsync();
// InvalidOperationException: owned entity cannot be queried independently

var lines = await context.Orders
    .SelectMany(o => o.OrderLines)
    .Where(l => l.Quantity > 5)
    .ToListAsync();
`

### 6.2 — Owned Entities Cannot Have Their Own PK

EF Core generates a shadow composite PK (owner FK + Id). You cannot define your own.

`csharp
line.HasKey(l => l.LineId); // Ignored for owned entities
`

### 6.3 — Include Requirement

OwnsOne types are auto-included (same row). OwnsMany requires .Include().

`csharp
var customer = await context.Customers.FirstAsync();
// customer.ShippingAddress is already populated (OwnsOne)

var order = await context.Orders
    .Include(o => o.OrderLines) // Required for OwnsMany
    .FirstAsync();
`

### 6.4 — Value Object Equality

Override Equals and GetHashCode for value semantics:

`csharp
public class Address : IEquatable<Address>
{
    public string Street { get; set; }
    public string City { get; set; }
    public string State { get; set; }
    public string ZipCode { get; set; }
    public string Country { get; set; }

    public bool Equals(Address other)
    {
        if (other is null) return false;
        return Street == other.Street && City == other.City
            && State == other.State && ZipCode == other.ZipCode
            && Country == other.Country;
    }
    public override bool Equals(object obj) => Equals(obj as Address);
    public override int GetHashCode() => HashCode.Combine(Street, City, State, ZipCode, Country);
    public static bool operator ==(Address left, Address right) => Equals(left, right);
    public static bool operator !=(Address left, Address right) => !Equals(left, right);
}
`

### 6.5 — Replace vs Mutate

Replacing the entire value object marks all columns as modified. Mutating individual properties only changes those columns.

`csharp
customer.ShippingAddress = new Address { ... };
// UPDATE all Shipping columns

customer.ShippingAddress.City = "Chicago";
// UPDATE ShippingCity only
`

### 6.6 — Nested Owned Types

`csharp

`

## 7 — Best Practices

### 7.1 — Prefer Table Splitting for Single Value Objects

Use OwnsOne with table splitting for single value objects (avoid JOINs). Use OwnsMany with separate tables for collections.

### 7.2 — Make Value Objects Immutable

`csharp
public class EmailAddress
{
    public string Value { get; }
    private EmailAddress() { }
    public EmailAddress(string value)
    {
        Value = value ?? throw new ArgumentNullException(nameof(value));
    }
}
`

### 7.3 — Override Equals for All Value Objects

Always implement IEquatable<T> for value objects to enable value equality semantics.

### 7.4 — Use Private Parameterless Constructors

EF Core needs a parameterless constructor. Make it private to preserve immutability:

`csharp
public class Money
{
    public decimal Amount { get; private set; }
    public string Currency { get; private set; }
    private Money() { }
    public Money(decimal amount, string currency)
    {
        Amount = amount;
        Currency = currency ?? throw new ArgumentNullException(nameof(currency));
    }
}
`

### 7.5 — Avoid Cyclic References

Owned types cannot reference their owner. Use regular entities (with identity) for bidirectional navigation.

### 7.6 — Dapper Custom Type Handler

`csharp
public class AddressTypeHandler : SqlMapper.ITypeHandler
{
    public object Parse(Type destinationType, object value)
    {
        var parts = value.ToString().Split('|');
        return new Address { Street = parts[0], City = parts[1], State = parts[2], ZipCode = parts[3], Country = parts[4] };
    }
    public void SetValue(IDbDataParameter parameter, object value)
    {
        var a = (Address)value;
        parameter.Value = $"{a.Street}|{a.City}|{a.State}|{a.ZipCode}|{a.Country}";
    }
}
SqlMapper.AddTypeHandler(new AddressTypeHandler());
`

## Appendix A — Immutable Address with Factory

`csharp
public class Address : IEquatable<Address>
{
    public string Street { get; }
    public string City { get; }
    public string State { get; }
    public string ZipCode { get; }
    public string Country { get; }
    private Address() { }
    public Address(string street, string city, string state, string zipCode, string country)
    {
        Street = street ?? throw new ArgumentNullException(nameof(street));
        City = city ?? throw new ArgumentNullException(nameof(city));
        State = state ?? throw new ArgumentNullException(nameof(state));
        ZipCode = zipCode ?? throw new ArgumentNullException(nameof(zipCode));
        Country = country ?? throw new ArgumentNullException(nameof(country));
    }
    public override string ToString() => $"{Street}, {City}, {State} {ZipCode}, {Country}";
    public bool Equals(Address other)
    {
        if (other is null) return false;
        if (ReferenceEquals(this, other)) return true;
        return Street == other.Street && City == other.City && State == other.State
            && ZipCode == other.ZipCode && Country == other.Country;
    }
    public override bool Equals(object obj) => Equals(obj as Address);
    public override int GetHashCode() => HashCode.Combine(Street, City, State, ZipCode, Country);
    public static bool operator ==(Address left, Address right) => Equals(left, right);
    public static bool operator !=(Address left, Address right) => !Equals(left, right);
}
`

## Appendix B — Owned Configuration by Convention

`csharp
public static class OwnedEntityConventions
{
    public static void ConfigureOwnedTypes(this ModelBuilder builder)
    {
        foreach (var entity in builder.Model.GetEntityTypes())
        {
            foreach (var navigation in entity.GetNavigations())
            {
                if (navigation.TargetEntityType.IsOwned())
                {
                    // Apply conventions to all owned types
                }
            }
        }
    }

    public static void AddOwned<TEntity, TOwned>(this EntityTypeBuilder<TEntity> builder,
        Expression<Func<TEntity, TOwned>> expression, string tableName)
        where TEntity : class
        where TOwned : class
    {
        builder.OwnsOne(expression, owned =>
        {
            owned.ToTable(tableName);
        });
    }
}

// Usage
builder.Entity<Customer>().AddOwned(c => c.ShippingAddress, "CustomerAddresses");
`

## Appendix D — Fluent Mapping with Column Prefix

`csharp
public static void MapValueObject<T>(this OwnsNavigationBuilder builder,
    Expression<Func<T>> navigation, string prefix)
    where T : class
{
    builder.OwnsOne(navigation, obj =>
    {
        foreach (var prop in typeof(T).GetProperties())
        {
            obj.Property(prop.Name).HasColumnName($"{prefix}{prop.Name}");
        }
    });
}

## Appendix E — Dapper Query for Owned Collection (Multiple Rows)

`csharp
public async Task<IEnumerable<Order>> GetAllOrdersWithLinesAsync()
{
    const string sql = @"
        SELECT o.OrderId, o.OrderDate, o.TotalAmount,
               ol.Id AS LineId, ol.ProductName, ol.Quantity, ol.UnitPrice
        FROM Orders o
        LEFT JOIN OrderLines ol ON o.OrderId = ol.OrderId
        ORDER BY o.OrderId, ol.Id";

    await using var conn = new SqlConnection(_connectionString);

    var orderDict = new Dictionary<int, Order>();

    await conn.QueryAsync<Order, OrderLine, Order>(sql, (order, line) =>
    {
        if (!orderDict.TryGetValue(order.OrderId, out var existing))
        {
            existing = order;
            existing.OrderLines = new List<OrderLine>();
            orderDict.Add(order.OrderId, existing);
        }
        if (line != null) existing.OrderLines.Add(line);
        return existing;
    }, splitOn: "LineId");

    return orderDict.Values;
}
`

## Appendix F — Value Object Validation

`csharp
public class EmailAddress
{
    public string Value { get; }
    private EmailAddress() { }
    public EmailAddress(string value)
    {
        if (string.IsNullOrWhiteSpace(value))
            throw new ArgumentException("Email is required", nameof(value));
        if (!value.Contains('@'))
            throw new ArgumentException("Email must contain @", nameof(value));
        Value = value;
    }
    public bool Equals(EmailAddress other) => other != null && Value == other.Value;
    public override bool Equals(object obj) => Equals(obj as EmailAddress);
    public override int GetHashCode() => Value.GetHashCode();
    public override string ToString() => Value;
}
`

## Appendix G — Table Per Hierarchy with Owned Types

`csharp
public abstract class Payment
{
    public int PaymentId { get; set; }
    public Money Amount { get; set; }
    public DateTime PaymentDate { get; set; }
}

public class CreditCardPayment : Payment
{
    public string LastFourDigits { get; set; }
}

public class BankTransferPayment : Payment
{
    public string BankAccountNumber { get; set; }
}

// Configure owned type on base entity
modelBuilder.Entity<Payment>(entity =>
{
    entity.HasKey(p => p.PaymentId);
    entity.OwnsOne(p => p.Amount, money =>
    {
        money.Property(m => m.Amount).HasColumnName("Amount");
        money.Property(m => m.Currency).HasColumnName("Currency");
    });
    // TPH mapping
    entity.HasDiscriminator<string>("PaymentType")
        .HasValue<CreditCardPayment>("CC")
        .HasValue<BankTransferPayment>("BT");
});
`

## Appendix H — Migration Script for Owned Types

-- Add OwnsOne columns to existing table
ALTER TABLE Customers
    ADD ShippingStreet NVARCHAR(200) NULL,
        ShippingCity NVARCHAR(100) NULL,
        ShippingState NVARCHAR(50) NULL,
        ShippingZipCode NVARCHAR(20) NULL,
        ShippingCountry NVARCHAR(100) NULL;

-- Add OwnsMany table
CREATE TABLE OrderLines (
    OrderId INT NOT NULL,
    Id INT IDENTITY(1,1) NOT NULL,
    ProductName NVARCHAR(200) NOT NULL,
    Quantity INT NOT NULL,
    UnitPrice DECIMAL(18,2) NOT NULL,
    CONSTRAINT PK_OrderLines PRIMARY KEY (OrderId, Id),
    CONSTRAINT FK_OrderLines_Orders FOREIGN KEY (OrderId) REFERENCES Orders(OrderId)
);

-- Migrate data to owned columns
UPDATE Customers
SET ShippingStreet = a.Street,
    ShippingCity = a.City,
    ShippingState = a.State,

    ShippingCountry = a.Country

## Appendix I — Comparing EF Core and Dapper for Owned Types

| Operation | EF Core | Dapper |
|-----------|---------|--------|
