## 52.01 — Middleware Pipeline Internals

**What middleware really is**
```csharp
// Middleware is a function: (HttpContext, next) → Task
// The pipeline is a chain of these functions

// Manual registration (using Use)
app.Use(async (HttpContext context, RequestDelegate next) =>
{
    // Before downstream
    context.Items["StartTime"] = DateTime.UtcNow;
    
    await next(context);  // Call next middleware
    
    // After downstream
    var elapsed = DateTime.UtcNow - (DateTime)context.Items["StartTime"]!;
    context.Response.Headers.Append("X-Response-Time", elapsed.TotalMilliseconds.ToString("F2"));
});

// Terminal middleware (no next call)
app.Run(async context =>
{
    await context.Response.WriteAsync("End of pipeline");
});

// Class-based middleware
public class RequestTimingMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger<RequestTimingMiddleware> _logger;
    
    public RequestTimingMiddleware(RequestDelegate next, ILogger<RequestTimingMiddleware> logger)
    {
        _next = next;
        _logger = logger;
    }
    
    public async Task InvokeAsync(HttpContext context)
    {
        var sw = Stopwatch.StartNew();
        try
        {
            await _next(context);
        }
        finally
        {
            sw.Stop();
            _logger.LogInformation(
                "{Method} {Path} responded {StatusCode} in {ElapsedMs}ms",
                context.Request.Method,
                context.Request.Path,
                context.Response.StatusCode,
                sw.ElapsedMilliseconds);
        }
    }
}

// Extension method for clean registration
public static IApplicationBuilder UseRequestTiming(this IApplicationBuilder app)
    => app.UseMiddleware<RequestTimingMiddleware>();

app.UseRequestTiming();
```

**Middleware order matters**
```csharp
// Correct order in ASP.NET Core:
app.UseExceptionHandler();       // 1. Catch exceptions from everything below
app.UseHsts();                   // 2. Security header
app.UseHttpsRedirection();       // 3. Force HTTPS
app.UseStaticFiles();            // 4. Short-circuit for static files
app.UseRouting();                // 5. Match routes (sets RouteData)
app.UseCors();                   // 6. After routing, before auth
app.UseAuthentication();         // 7. Who is the user?
app.UseAuthorization();          // 8. Is the user allowed? (MUST be after auth)
app.UseOutputCache();            // 9. After auth (cached responses respect auth)
app.UseRateLimiter();            // 10. After auth (rate limit per user)
app.MapControllers();            // 11. Endpoint execution
```

## 52.02 — Dependency Injection Deep Dive

**Service lifetimes and rules**
```csharp
// Singleton: created once, shared for app lifetime
// Scoped: created once per HTTP request (or manually created scope)
// Transient: created every time requested

// RULE: Never inject a shorter-lived service into a longer-lived one
// Singleton → Scoped: CAPTIVE DEPENDENCY (bug! scoped leaks into singleton)
// Scoped → Transient: OK (transient gets created per-request anyway)
// Transient → Singleton: OK (singleton holds transient forever — potential memory leak)

// Detect captive dependencies:
builder.Services.AddScoped<ICurrentUserService, CurrentUserService>();
// If OrderService is Singleton and injects ICurrentUserService → BUG
// The singleton captures the first request's user service

// Solution: inject IServiceScopeFactory for on-demand scope creation
public class BackgroundJobService
{
    private readonly IServiceScopeFactory _scopeFactory;
    
    public async Task RunJobAsync(CancellationToken ct)
    {
        await using var scope = _scopeFactory.CreateAsyncScope();
        var db = scope.ServiceProvider.GetRequiredService<AppDbContext>();  // Scoped, new per call
        var currentUser = scope.ServiceProvider.GetRequiredService<ICurrentUserService>();
        // Use them safely within this scope
    }
}
```

**Keyed services (.NET 8)**
```csharp
// Register multiple implementations of the same interface by key
builder.Services.AddKeyedSingleton<IPaymentProcessor, StripeProcessor>("stripe");
builder.Services.AddKeyedSingleton<IPaymentProcessor, PayPalProcessor>("paypal");
builder.Services.AddKeyedSingleton<IPaymentProcessor, CreditCardProcessor>("creditcard");

// Resolve by key
public class PaymentService([FromKeyedServices("stripe")] IPaymentProcessor stripe,
                            [FromKeyedServices("paypal")] IPaymentProcessor paypal)
{
    public async Task ProcessAsync(PaymentRequest request)
    {
        var processor = request.Method switch
        {
            "stripe" => stripe,
            "paypal" => paypal,
            _ => throw new NotSupportedException()
        };
        await processor.ProcessAsync(request);
    }
}

// Or resolve dynamically
var processor = serviceProvider.GetRequiredKeyedService<IPaymentProcessor>("stripe");
```

**Decorator pattern with DI**
```csharp
// Add cross-cutting concerns (logging, caching, validation) without modifying original

public class CachingProductRepository : IProductRepository
{
    private readonly IProductRepository _inner;  // Decorated service
    private readonly IMemoryCache _cache;
    
    public CachingProductRepository(IProductRepository inner, IMemoryCache cache)
    {
        _inner = inner;
        _cache = cache;
    }
    
    public async Task<Product?> GetByIdAsync(int id, CancellationToken ct)
    {
        return await _cache.GetOrCreateAsync($"product:{id}", async entry =>
        {
            entry.AbsoluteExpirationRelativeToNow = TimeSpan.FromMinutes(10);
            return await _inner.GetByIdAsync(id, ct);
        });
    }
}

// Registration with Scrutor (decorator helper)
// NuGet: Scrutor
builder.Services.AddScoped<IProductRepository, SqlProductRepository>();
builder.Services.Decorate<IProductRepository, CachingProductRepository>();
// Now IProductRepository resolves to CachingProductRepository(SqlProductRepository)
```

**Open generic registrations**
```csharp
// Register IRepository<> so any IRepository<T> resolves to Repository<T>
builder.Services.AddScoped(typeof(IRepository<>), typeof(Repository<>));

// Now:
services.GetRequiredService<IRepository<Order>>();   // → Repository<Order>
services.GetRequiredService<IRepository<Product>>(); // → Repository<Product>
```

## 52.03 — Authentication Deep Dive

**Cookie authentication (server-rendered apps)**
```csharp
builder.Services.AddAuthentication(CookieAuthenticationDefaults.AuthenticationScheme)
    .AddCookie(options =>
    {
        options.Cookie.Name = "myapp-auth";
        options.Cookie.HttpOnly = true;      // Not accessible to JavaScript
        options.Cookie.SameSite = SameSiteMode.Strict;  // CSRF protection
        options.Cookie.SecurePolicy = CookieSecurePolicy.Always;  // HTTPS only
        options.ExpireTimeSpan = TimeSpan.FromHours(8);
        options.SlidingExpiration = true;   // Reset expiry on each request
        options.LoginPath = "/login";
        options.LogoutPath = "/logout";
        options.AccessDeniedPath = "/access-denied";
        
        // For API: return 401/403 instead of redirecting
        options.Events.OnRedirectToLogin = ctx =>
        {
            if (ctx.Request.Path.StartsWithSegments("/api"))
            {
                ctx.Response.StatusCode = StatusCodes.Status401Unauthorized;
                return Task.CompletedTask;
            }
            ctx.Response.Redirect(ctx.RedirectUri);
            return Task.CompletedTask;
        };
    });

// Sign in
var claims = new List<Claim>
{
    new(ClaimTypes.NameIdentifier, user.Id.ToString()),
    new(ClaimTypes.Email, user.Email),
    new(ClaimTypes.Role, user.Role)
};
var identity = new ClaimsIdentity(claims, CookieAuthenticationDefaults.AuthenticationScheme);
var principal = new ClaimsPrincipal(identity);

await HttpContext.SignInAsync(
    CookieAuthenticationDefaults.AuthenticationScheme,
    principal,
    new AuthenticationProperties { IsPersistent = rememberMe });
```

**Multiple authentication schemes**
```csharp
// API: JWT for client apps, API keys for service-to-service
builder.Services.AddAuthentication()
    .AddJwtBearer("jwt", options => { /* JWT config */ })
    .AddScheme<ApiKeyAuthenticationOptions, ApiKeyAuthenticationHandler>("apikey", options => { });

// Endpoint-specific scheme
app.MapGet("/api/data", GetData)
    .RequireAuthorization(policy => policy.AddAuthenticationSchemes("jwt", "apikey"));

// Default policy combining multiple schemes
builder.Services.AddAuthorization(options =>
{
    options.DefaultPolicy = new AuthorizationPolicyBuilder()
        .AddAuthenticationSchemes("jwt", "apikey")
        .RequireAuthenticatedUser()
        .Build();
});
```
