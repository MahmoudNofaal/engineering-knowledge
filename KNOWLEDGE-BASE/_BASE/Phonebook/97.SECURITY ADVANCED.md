## 97.01 — OWASP Top 10 (2023) — .NET Implementation

**A01: Broken Access Control**
```csharp
// WRONG: Trust the client's user ID from the request body
[HttpPost("transfer")]
public async Task<IActionResult> Transfer(TransferRequest request)
{
    await _bank.TransferAsync(request.FromAccountId, request.ToAccountId, request.Amount);
    // Anyone can transfer from any account!
}

// RIGHT: Get user ID from the authenticated token
[HttpPost("transfer"), Authorize]
public async Task<IActionResult> Transfer(TransferRequest request)
{
    var userId = int.Parse(User.FindFirstValue(ClaimTypes.NameIdentifier)!);
    var account = await _accounts.GetByIdAsync(request.FromAccountId);
    
    if (account.OwnerId != userId)
        return Forbid();  // Can only transfer from OWN accounts
    
    await _bank.TransferAsync(request.FromAccountId, request.ToAccountId, request.Amount);
    return Ok();
}
```

**A02: Cryptographic Failures**
```csharp
// WRONG: Sensitive data unencrypted
var user = await _db.Users.Where(u => u.SSN == ssn).FirstOrDefaultAsync();
// SSN stored in plain text in database!

// RIGHT: Deterministic encryption for searchable fields
public class EncryptedField
{
    private readonly byte[] _key;
    
    public string Encrypt(string plaintext)
    {
        using var aes = Aes.Create();
        aes.Key = _key;
        aes.GenerateIV();
        
        using var encryptor = aes.CreateEncryptor();
        var plaintextBytes = Encoding.UTF8.GetBytes(plaintext);
        var encrypted = encryptor.TransformFinalBlock(plaintextBytes, 0, plaintextBytes.Length);
        
        // Prepend IV to ciphertext
        var result = new byte[aes.IV.Length + encrypted.Length];
        Buffer.BlockCopy(aes.IV, 0, result, 0, aes.IV.Length);
        Buffer.BlockCopy(encrypted, 0, result, aes.IV.Length, encrypted.Length);
        
        return Convert.ToBase64String(result);
    }
    
    // For searchable: HMAC the value (deterministic, one-way)
    public string Hash(string plaintext)
    {
        using var hmac = new HMACSHA256(_key);
        return Convert.ToBase64String(hmac.ComputeHash(Encoding.UTF8.GetBytes(plaintext)));
    }
}

// Store: hash_ssn = Hash(ssn), encrypted_ssn = Encrypt(ssn)
// Search: WHERE hash_ssn = Hash(inputSsn)
// Display: Decrypt(encrypted_ssn)

// WRONG: Weak password hashing
var hash = MD5.HashData(Encoding.UTF8.GetBytes(password));  // NEVER

// RIGHT: bcrypt/Argon2 via ASP.NET Core PasswordHasher
var hasher = new PasswordHasher<User>();
var hash = hasher.HashPassword(user, password);           // bcrypt internally
var result = hasher.VerifyHashedPassword(user, hash, password);  // Verify
```

**A07: Identification and Authentication Failures**
```csharp
// Account lockout — prevent brute force
public async Task<LoginResult> LoginAsync(LoginRequest request, CancellationToken ct)
{
    var user = await _users.GetByEmailAsync(request.Email, ct);
    if (user is null)
    {
        // Constant-time response to prevent user enumeration
        await Task.Delay(Random.Shared.Next(100, 200), ct);
        return LoginResult.Failure("Invalid credentials");
    }
    
    if (user.LockedUntil > DateTime.UtcNow)
        return LoginResult.Failure($"Account locked until {user.LockedUntil:g}");
    
    if (!VerifyPassword(user, request.Password))
    {
        user.FailedLoginCount++;
        if (user.FailedLoginCount >= 5)
            user.LockedUntil = DateTime.UtcNow.AddMinutes(15);
        
        await _users.UpdateAsync(user, ct);
        return LoginResult.Failure("Invalid credentials");
    }
    
    user.FailedLoginCount = 0;
    user.LastLoginAt = DateTime.UtcNow;
    await _users.UpdateAsync(user, ct);
    
    return LoginResult.Success(GenerateTokens(user));
}

// Secure password reset flow
public async Task<string> RequestPasswordResetAsync(string email, CancellationToken ct)
{
    var user = await _users.GetByEmailAsync(email, ct);
    
    // Always return same response (prevent email enumeration)
    if (user is null) return "If this email exists, a reset link was sent";
    
    // Generate secure token (not guessable)
    var token = Convert.ToBase64String(RandomNumberGenerator.GetBytes(32));
    
    // Store hash of token (never store plain tokens)
    user.PasswordResetTokenHash = SHA256.HashData(Encoding.UTF8.GetBytes(token)).ToHexString();
    user.PasswordResetTokenExpiry = DateTime.UtcNow.AddHours(1);
    
    await _users.UpdateAsync(user, ct);
    await _email.SendPasswordResetAsync(email, token, ct);
    
    return "If this email exists, a reset link was sent";
}
```
