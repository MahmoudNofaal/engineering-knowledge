## 86.01 — Thread Synchronization Primitives

**SemaphoreSlim — limit concurrent access**
```csharp
// Allow at most N concurrent operations
private readonly SemaphoreSlim _semaphore = new(initialCount: 3, maxCount: 3);

public async Task<string> GetDataAsync(string key, CancellationToken ct)
{
    await _semaphore.WaitAsync(ct);  // Blocks if 3 already running
    try
    {
        return await FetchFromServiceAsync(key, ct);
    }
    finally
    {
        _semaphore.Release();  // Always release, even on exception
    }
}

// Rate limiter: limit to N operations per second
public async Task EnqueueWorkAsync(Func<Task> work, CancellationToken ct)
{
    await _semaphore.WaitAsync(ct);
    _ = Task.Run(async () =>
    {
        try { await work(); }
        finally
        {
            await Task.Delay(TimeSpan.FromSeconds(1), ct);  // Release after 1 second
            _semaphore.Release();
        }
    }, ct);
}
```

**ReaderWriterLockSlim — many readers, one writer**
```csharp
private readonly ReaderWriterLockSlim _lock = new();
private Dictionary<string, CachedItem> _cache = new();

// Multiple threads can read simultaneously
public CachedItem? Read(string key)
{
    _lock.EnterReadLock();
    try
    {
        return _cache.TryGetValue(key, out var item) ? item : null;
    }
    finally
    {
        _lock.ExitReadLock();
    }
}

// Only one thread can write (blocks all readers and other writers)
public void Write(string key, CachedItem item)
{
    _lock.EnterWriteLock();
    try
    {
        _cache[key] = item;
    }
    finally
    {
        _lock.ExitWriteLock();
    }
}

// Upgradeable read lock (read first, promote to write if needed)
public CachedItem GetOrCreate(string key, Func<CachedItem> factory)
{
    _lock.EnterUpgradeableReadLock();
    try
    {
        if (_cache.TryGetValue(key, out var existing))
            return existing;
        
        _lock.EnterWriteLock();
        try
        {
            // Double-check after acquiring write lock
            if (_cache.TryGetValue(key, out existing))
                return existing;
            
            var item = factory();
            _cache[key] = item;
            return item;
        }
        finally
        {
            _lock.ExitWriteLock();
        }
    }
    finally
    {
        _lock.ExitUpgradeableReadLock();
    }
}
```

**Channels — async producer-consumer**
```csharp
// Channel<T>: high-performance async queue
// Better than BlockingCollection for async code

// Bounded channel: back-pressure when consumer is slow
var channel = Channel.CreateBounded<WorkItem>(new BoundedChannelOptions(capacity: 1000)
{
    FullMode = BoundedChannelFullMode.Wait,  // Producer awaits when full
    SingleWriter = false,
    SingleReader = false
});

var writer = channel.Writer;
var reader = channel.Reader;

// Producer
async Task ProduceAsync(CancellationToken ct)
{
    try
    {
        while (!ct.IsCancellationRequested)
        {
            var item = await GetNextWorkItemAsync(ct);
            await writer.WriteAsync(item, ct);  // Backpressure: awaits if full
        }
    }
    finally
    {
        writer.Complete();  // Signal no more items
    }
}

// Consumer
async Task ConsumeAsync(CancellationToken ct)
{
    await foreach (var item in reader.ReadAllAsync(ct))
    {
        await ProcessAsync(item, ct);
    }
}

// Multiple consumers (parallel processing)
async Task RunWithParallelConsumersAsync(int consumerCount, CancellationToken ct)
{
    var producer = ProduceAsync(ct);
    var consumers = Enumerable.Range(0, consumerCount)
        .Select(_ => ConsumeAsync(ct));
    
    await Task.WhenAll(consumers.Append(producer));
}
```

## 86.02 — Task Parallel Library (TPL) Advanced

**Dataflow blocks (pipeline processing)**
```csharp
// NuGet: System.Threading.Tasks.Dataflow
// Build processing pipelines with bounded parallelism

var downloadBlock = new TransformBlock<string, byte[]>(
    async url => await _http.GetByteArrayAsync(url),
    new ExecutionDataflowBlockOptions { MaxDegreeOfParallelism = 4 });

var parseBlock = new TransformBlock<byte[], ParsedData>(
    bytes => ParseData(bytes),
    new ExecutionDataflowBlockOptions { MaxDegreeOfParallelism = 2 });

var saveBlock = new ActionBlock<ParsedData>(
    async data => await SaveToDbAsync(data),
    new ExecutionDataflowBlockOptions { MaxDegreeOfParallelism = 1 });

// Link blocks into pipeline
downloadBlock.LinkTo(parseBlock, new DataflowLinkOptions { PropagateCompletion = true });
parseBlock.LinkTo(saveBlock, new DataflowLinkOptions { PropagateCompletion = true });

// Feed input
foreach (var url in urls)
    await downloadBlock.SendAsync(url);

downloadBlock.Complete();        // Signal no more input
await saveBlock.Completion;      // Wait for all to complete
```

**PLINQ (Parallel LINQ)**
```csharp
// Parallel query execution
var results = largeCollection
    .AsParallel()                          // Enable parallel processing
    .WithDegreeOfParallelism(4)            // Max 4 threads
    .WithCancellation(ct)                  // Support cancellation
    .Where(item => IsExpensive(item))      // Runs in parallel
    .Select(item => Transform(item))       // Runs in parallel
    .ToList();                             // Materialize result

// Ordered parallel query (maintains order, less efficient)
var orderedResults = largeCollection
    .AsParallel()
    .AsOrdered()  // Preserve source order in result
    .Select(item => Transform(item))
    .ToList();

// Force sequential for part of the query
var mixed = largeCollection
    .AsParallel()
    .Where(expensive)     // Parallel
    .AsSequential()       // Switch to sequential
    .Where(cheap)         // Sequential (e.g., has side effects)
    .ToList();

// When NOT to use PLINQ:
// - Short-running operations (parallelism overhead > benefit)
// - Operations with significant synchronization (lock contention kills parallelism)
// - Database queries (use async, not parallel)
```
