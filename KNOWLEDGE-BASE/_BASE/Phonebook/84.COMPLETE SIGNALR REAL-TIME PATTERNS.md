## 84.01 — SignalR Hub Advanced Patterns

**Typed Hub — compile-time safety**
```csharp
// Define what the server can call on the client
public interface IOrderHubClient
{
    Task OrderStatusChanged(int orderId, string newStatus, DateTime changedAt);
    Task NewMessageReceived(ChatMessage message);
    Task UserJoinedRoom(string roomName, string username);
    Task UserLeftRoom(string roomName, string username);
    Task ReceiveNotification(NotificationDto notification);
}

// Typed hub — Clients.All is now IOrderHubClient
public class OrderHub : Hub<IOrderHubClient>
{
    private readonly IOrderRepository _orders;
    
    public async Task SubscribeToOrder(int orderId)
    {
        var order = await _orders.GetByIdAsync(orderId, CancellationToken.None);
        if (order is null) return;
        
        // Add to group named "order:42"
        await Groups.AddToGroupAsync(Context.ConnectionId, $"order:{orderId}");
    }
    
    public async Task UnsubscribeFromOrder(int orderId)
        => await Groups.RemoveFromGroupAsync(Context.ConnectionId, $"order:{orderId}");
    
    public async Task JoinRoom(string roomName)
    {
        await Groups.AddToGroupAsync(Context.ConnectionId, $"room:{roomName}");
        
        // Notify others in room — COMPILE-TIME CHECKED method name
        await Clients.OthersInGroup($"room:{roomName}")
            .UserJoinedRoom(roomName, Context.User?.Identity?.Name ?? "Anonymous");
    }
    
    public override async Task OnDisconnectedAsync(Exception? ex)
    {
        // Clean up user state on disconnect
        var userId = Context.UserIdentifier;
        if (userId is not null)
            await _presenceService.UserDisconnectedAsync(userId);
        
        await base.OnDisconnectedAsync(ex);
    }
}

// Push updates from server-side services (background jobs, domain events)
public class OrderStatusChangedHandler : INotificationHandler<OrderStatusChangedEvent>
{
    private readonly IHubContext<OrderHub, IOrderHubClient> _hub;
    
    public async Task Handle(OrderStatusChangedEvent evt, CancellationToken ct)
    {
        // Push to all connections subscribed to this order
        await _hub.Clients
            .Group($"order:{evt.OrderId}")
            .OrderStatusChanged(evt.OrderId, evt.NewStatus, evt.ChangedAt);
    }
}
```

**User-specific notifications (per-connection)**
```csharp
// SignalR maps User.Identity.Name to all connections for that user
// Requires authentication

builder.Services.AddSignalR();
// User ID extraction from JWT claims:
builder.Services.AddSingleton<IUserIdProvider, UserIdProvider>();

public class UserIdProvider : IUserIdProvider
{
    public string? GetUserId(HubConnectionContext connection)
        => connection.User?.FindFirstValue(ClaimTypes.NameIdentifier);
}

// Now Clients.User("user-id-42") sends to all devices logged in as user 42
public class NotificationService
{
    private readonly IHubContext<OrderHub, IOrderHubClient> _hub;
    
    public async Task NotifyUserAsync(string userId, NotificationDto notification)
        => await _hub.Clients.User(userId).ReceiveNotification(notification);
    
    public async Task NotifyAllAsync(NotificationDto notification)
        => await _hub.Clients.All.ReceiveNotification(notification);
    
    public async Task NotifyGroupAsync(string groupName, NotificationDto notification)
        => await _hub.Clients.Group(groupName).ReceiveNotification(notification);
}
```

**Presence tracking**
```csharp
public class PresenceTracker
{
    private readonly IDistributedCache _cache;
    private const string OnlineUsersKey = "signalr:online-users";
    
    public async Task UserConnectedAsync(string userId, string connectionId)
    {
        var connections = await GetConnectionsAsync(userId);
        connections.Add(connectionId);
        await SaveConnectionsAsync(userId, connections);
        
        // Mark user online if this is their first connection
        if (connections.Count == 1)
            await _cache.SetStringAsync($"user:{userId}:online", "1",
                new DistributedCacheEntryOptions { SlidingExpiration = TimeSpan.FromHours(24) });
    }
    
    public async Task UserDisconnectedAsync(string userId, string connectionId)
    {
        var connections = await GetConnectionsAsync(userId);
        connections.Remove(connectionId);
        
        if (connections.Count == 0)
        {
            // User went offline (all tabs/devices closed)
            await _cache.RemoveAsync($"user:{userId}:online");
            await SaveConnectionsAsync(userId, connections);
        }
        else
            await SaveConnectionsAsync(userId, connections);
    }
    
    public async Task<bool> IsOnlineAsync(string userId)
        => await _cache.GetStringAsync($"user:{userId}:online") is not null;
}
```
