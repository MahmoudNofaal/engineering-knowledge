## 47.01 — Repository Pattern (Complete)

```csharp
// Generic interface
public interface IRepository<T, TId> where T : class
{
    Task<T?> GetByIdAsync(TId id, CancellationToken ct = default);
    Task<IReadOnlyList<T>> GetAllAsync(CancellationToken ct = default);
    Task<IReadOnlyList<T>> FindAsync(Expression<Func<T, bool>> predicate, CancellationToken ct = default);
    Task<bool> ExistsAsync(TId id, CancellationToken ct = default);
    Task<int> CountAsync(Expression<Func<T, bool>>? predicate = null, CancellationToken ct = default);
    void Add(T entity);
    void Update(T entity);
    void Remove(T entity);
}

// Generic implementation
public abstract class Repository<T, TId> : IRepository<T, TId>
    where T : class
{
    protected readonly AppDbContext Db;
    
    protected Repository(AppDbContext db) => Db = db;
    
    public virtual async Task<T?> GetByIdAsync(TId id, CancellationToken ct = default)
        => await Db.Set<T>().FindAsync(new object?[] { id }, ct);
    
    public virtual async Task<IReadOnlyList<T>> GetAllAsync(CancellationToken ct = default)
        => await Db.Set<T>().ToListAsync(ct);
    
    public virtual async Task<IReadOnlyList<T>> FindAsync(
        Expression<Func<T, bool>> predicate, CancellationToken ct = default)
        => await Db.Set<T>().Where(predicate).ToListAsync(ct);
    
    public virtual async Task<bool> ExistsAsync(TId id, CancellationToken ct = default)
        => await Db.Set<T>().FindAsync(new object?[] { id }, ct) is not null;
    
    public virtual async Task<int> CountAsync(
        Expression<Func<T, bool>>? predicate = null, CancellationToken ct = default)
        => predicate is null
            ? await Db.Set<T>().CountAsync(ct)
            : await Db.Set<T>().CountAsync(predicate, ct);
    
    public virtual void Add(T entity) => Db.Set<T>().Add(entity);
    public virtual void Update(T entity) => Db.Set<T>().Update(entity);
    public virtual void Remove(T entity) => Db.Set<T>().Remove(entity);
}

// Specific repository with custom queries
public class OrderRepository : Repository<Order, int>, IOrderRepository
{
    public OrderRepository(AppDbContext db) : base(db) { }
    
    public async Task<IReadOnlyList<Order>> GetByCustomerAsync(
        int customerId, CancellationToken ct = default)
        => await Db.Orders
            .Include(o => o.Items)
            .Where(o => o.CustomerId == customerId)
            .OrderByDescending(o => o.CreatedAt)
            .AsNoTracking()
            .ToListAsync(ct);
    
    public async Task<PagedResult<Order>> GetPagedAsync(
        int page, int pageSize, OrderFilter filter, CancellationToken ct = default)
    {
        var query = Db.Orders.AsNoTracking();
        
        if (filter.Status.HasValue)
            query = query.Where(o => o.Status == filter.Status);
        
        if (filter.FromDate.HasValue)
            query = query.Where(o => o.CreatedAt >= filter.FromDate);
        
        var totalCount = await query.CountAsync(ct);
        var items = await query
            .OrderByDescending(o => o.CreatedAt)
            .Skip((page - 1) * pageSize)
            .Take(pageSize)
            .ToListAsync(ct);
        
        return new PagedResult<Order>(items, totalCount, page, pageSize);
    }
}
```

## 47.02 — Pagination Pattern

```csharp
// Generic paged result
public record PagedResult<T>(
    IReadOnlyList<T> Items,
    int TotalCount,
    int Page,
    int PageSize)
{
    public int TotalPages => (int)Math.Ceiling((double)TotalCount / PageSize);
    public bool HasPreviousPage => Page > 1;
    public bool HasNextPage => Page < TotalPages;
    public bool IsFirstPage => Page == 1;
    public bool IsLastPage => Page == TotalPages;
}

// Extension method for EF Core
public static async Task<PagedResult<T>> ToPagedResultAsync<T>(
    this IQueryable<T> query,
    int page,
    int pageSize,
    CancellationToken ct = default)
{
    var totalCount = await query.CountAsync(ct);
    var items = await query
        .Skip((page - 1) * pageSize)
        .Take(pageSize)
        .ToListAsync(ct);
    
    return new PagedResult<T>(items, totalCount, page, pageSize);
}

// Cursor-based pagination (more efficient for large datasets)
public record CursorPagedResult<T>(
    IReadOnlyList<T> Items,
    string? NextCursor,  // Opaque cursor for next page
    string? PreviousCursor,
    bool HasMore);

public async Task<CursorPagedResult<Order>> GetOrdersWithCursorAsync(
    string? cursor, int pageSize, CancellationToken ct)
{
    var query = Db.Orders.AsNoTracking().OrderByDescending(o => o.Id);
    
    if (cursor is not null)
    {
        var lastId = int.Parse(Base64.Decode(cursor));
        query = (IOrderedQueryable<Order>)query.Where(o => o.Id < lastId);
    }
    
    var items = await query.Take(pageSize + 1).ToListAsync(ct);
    var hasMore = items.Count > pageSize;
    if (hasMore) items.RemoveAt(items.Count - 1);
    
    var nextCursor = hasMore ? Base64.Encode(items.Last().Id.ToString()) : null;
    return new CursorPagedResult<Order>(items, nextCursor, cursor, hasMore);
}
```

## 47.03 — Caching Pattern (Multi-Level)

```csharp
public class MultiLevelCacheService<T>
{
    private readonly IMemoryCache _l1;           // In-process, fastest
    private readonly IDistributedCache _l2;      // Redis, shared
    private readonly TimeSpan _l1Duration = TimeSpan.FromMinutes(1);
    private readonly TimeSpan _l2Duration = TimeSpan.FromMinutes(10);
    
    public async Task<T?> GetAsync(string key, Func<Task<T?>> factory, CancellationToken ct)
    {
        // L1: in-memory cache
        if (_l1.TryGetValue(key, out T? cached))
            return cached;
        
        // L2: distributed cache (Redis)
        var serialized = await _l2.GetStringAsync(key, ct);
        if (serialized is not null)
        {
            var fromRedis = JsonSerializer.Deserialize<T>(serialized);
            _l1.Set(key, fromRedis, _l1Duration);  // Warm L1
            return fromRedis;
        }
        
        // L3: source of truth (database/API)
        var value = await factory();
        if (value is not null)
        {
            var json = JsonSerializer.Serialize(value);
            await _l2.SetStringAsync(key, json,
                new DistributedCacheEntryOptions { AbsoluteExpirationRelativeToNow = _l2Duration },
                ct);
            _l1.Set(key, value, _l1Duration);
        }
        
        return value;
    }
    
    public async Task InvalidateAsync(string key, CancellationToken ct)
    {
        _l1.Remove(key);
        await _l2.RemoveAsync(key, ct);
    }
}
```

## 47.04 — Background Job Queue Pattern

```csharp
// Job definition
public abstract record BackgroundJob
{
    public Guid Id { get; init; } = Guid.NewGuid();
    public DateTime EnqueuedAt { get; init; } = DateTime.UtcNow;
    public int Attempt { get; init; } = 0;
    public int MaxAttempts { get; init; } = 3;
}

public record SendEmailJob(string To, string Subject, string Body) : BackgroundJob;
public record GenerateReportJob(int ReportId, int UserId) : BackgroundJob;

// Job processor
public interface IJobHandler<TJob> where TJob : BackgroundJob
{
    Task HandleAsync(TJob job, CancellationToken ct);
}

public class SendEmailJobHandler : IJobHandler<SendEmailJob>
{
    private readonly IEmailService _email;
    
    public SendEmailJobHandler(IEmailService email) => _email = email;
    
    public async Task HandleAsync(SendEmailJob job, CancellationToken ct)
        => await _email.SendAsync(job.To, job.Subject, job.Body, ct);
}

// Dispatcher (routes job to handler)
public class JobDispatcher
{
    private readonly IServiceProvider _sp;
    
    public async Task DispatchAsync<TJob>(TJob job, CancellationToken ct)
        where TJob : BackgroundJob
    {
        var handler = _sp.GetRequiredService<IJobHandler<TJob>>();
        
        for (int attempt = 0; attempt <= job.MaxAttempts; attempt++)
        {
            try
            {
                await handler.HandleAsync(job with { Attempt = attempt }, ct);
                return;
            }
            catch when (attempt < job.MaxAttempts)
            {
                var delay = TimeSpan.FromSeconds(Math.Pow(2, attempt));  // Exponential backoff
                await Task.Delay(delay, ct);
            }
        }
        throw new InvalidOperationException($"Job {job.Id} failed after {job.MaxAttempts} attempts");
    }
}
```

## 47.05 — Event-Driven Architecture Patterns

**Domain Events dispatch via MediatR**
```csharp
// Base classes
public abstract record DomainEvent
{
    public Guid Id { get; } = Guid.NewGuid();
    public DateTime OccurredOn { get; } = DateTime.UtcNow;
}

public abstract class AggregateRoot
{
    private readonly List<DomainEvent> _events = new();
    
    protected void Raise(DomainEvent @event) => _events.Add(@event);
    
    public IReadOnlyList<DomainEvent> GetEvents() => _events.AsReadOnly();
    public void ClearEvents() => _events.Clear();
}

// Interceptor dispatches events after SaveChanges
public class DomainEventsDispatcher
{
    private readonly IMediator _mediator;
    private readonly AppDbContext _db;
    
    public async Task DispatchAsync(CancellationToken ct)
    {
        var aggregates = _db.ChangeTracker
            .Entries<AggregateRoot>()
            .Select(e => e.Entity)
            .Where(a => a.GetEvents().Any())
            .ToList();
        
        var events = aggregates.SelectMany(a => a.GetEvents()).ToList();
        aggregates.ForEach(a => a.ClearEvents());
        
        await _db.SaveChangesAsync(ct);  // Save first
        
        // Then dispatch events — they see committed state
        foreach (var @event in events)
            await _mediator.Publish(@event, ct);
    }
}

// Domain event handlers
public class OrderShippedNotificationHandler : INotificationHandler<OrderShippedEvent>
{
    private readonly IEmailService _email;
    private readonly IOrderRepository _orders;
    
    public async Task Handle(OrderShippedEvent notification, CancellationToken ct)
    {
        var order = await _orders.GetByIdAsync(notification.OrderId, ct);
        await _email.SendShippingConfirmationAsync(
            order!.CustomerEmail, notification.TrackingNumber, ct);
    }
}

public class OrderShippedAnalyticsHandler : INotificationHandler<OrderShippedEvent>
{
    public async Task Handle(OrderShippedEvent notification, CancellationToken ct)
    {
        // Track analytics — runs concurrently with email handler
        await _analytics.TrackEventAsync("OrderShipped", new
        {
            notification.OrderId,
            notification.OccurredOn
        }, ct);
    }
}
```

## 47.06 — FluentValidation Patterns

```csharp
// NuGet: FluentValidation.AspNetCore

public class CreateOrderRequestValidator : AbstractValidator<CreateOrderRequest>
{
    private readonly IProductRepository _products;
    
    public CreateOrderRequestValidator(IProductRepository products)
    {
        _products = products;
        
        RuleFor(x => x.CustomerId)
            .GreaterThan(0)
            .WithMessage("Customer ID must be a positive integer");
        
        RuleFor(x => x.Items)
            .NotEmpty()
            .WithMessage("Order must have at least one item")
            .Must(items => items.Count <= 50)
            .WithMessage("Order cannot have more than 50 items");
        
        RuleForEach(x => x.Items).SetValidator(new OrderItemValidator());
        
        // Async rule — database lookup
        RuleFor(x => x.CustomerId)
            .MustAsync(async (id, ct) =>
            {
                return await _products.CustomerExistsAsync(id, ct);
            })
            .WithMessage("Customer not found")
            .When(x => x.CustomerId > 0);  // Only validate if basic validation passes
    }
}

public class OrderItemValidator : AbstractValidator<OrderItemDto>
{
    public OrderItemValidator()
    {
        RuleFor(x => x.ProductId).GreaterThan(0);
        RuleFor(x => x.Quantity).InclusiveBetween(1, 100);
        RuleFor(x => x.Price).GreaterThan(0).LessThanOrEqualTo(100000);
    }
}

// Registration (auto-discovers all validators in assembly)
builder.Services.AddValidatorsFromAssemblyContaining<CreateOrderRequestValidator>();

// Usage in minimal API
app.MapPost("/api/orders", async (
    CreateOrderRequest request,
    IValidator<CreateOrderRequest> validator,
    ISender mediator,
    CancellationToken ct) =>
{
    var validation = await validator.ValidateAsync(request, ct);
    if (!validation.IsValid)
        return Results.ValidationProblem(validation.ToDictionary());
    
    var result = await mediator.Send(new CreateOrderCommand(request), ct);
    return Results.Created($"/api/orders/{result.Id}", result);
});
```
