## 62.01 — Custom Authorization Requirements

**Complete example: Time-based access**
```csharp
// Requirement
public class BusinessHoursRequirement : IAuthorizationRequirement
{
    public TimeOnly Start { get; }
    public TimeOnly End { get; }
    
    public BusinessHoursRequirement(TimeOnly start, TimeOnly end)
    {
        Start = start;
        End = end;
    }
}

// Handler
public class BusinessHoursHandler : AuthorizationHandler<BusinessHoursRequirement>
{
    private readonly TimeProvider _time;
    
    public BusinessHoursHandler(TimeProvider time) => _time = time;
    
    protected override Task HandleRequirementAsync(
        AuthorizationHandlerContext context,
        BusinessHoursRequirement requirement)
    {
        var currentTime = TimeOnly.FromDateTime(_time.GetLocalNow().DateTime);
        
        if (currentTime >= requirement.Start && currentTime <= requirement.End)
            context.Succeed(requirement);
        else
            context.Fail(new AuthorizationFailureReason(this,
                $"Access only allowed between {requirement.Start} and {requirement.End}"));
        
        return Task.CompletedTask;
    }
}

// Policy
builder.Services.AddAuthorization(options =>
{
    options.AddPolicy("BusinessHours", policy =>
        policy.AddRequirements(
            new BusinessHoursRequirement(
                new TimeOnly(9, 0),   // 9 AM
                new TimeOnly(17, 0)))); // 5 PM
});
```

**Endpoint-level fine-grained authorization**
```csharp
// Custom authorization attribute using IAuthorizationFilter
public class RequiresOrderOwnershipAttribute : Attribute, IAsyncAuthorizationFilter
{
    public async Task OnAuthorizationAsync(AuthorizationFilterContext context)
    {
        var userId = int.Parse(context.HttpContext.User
            .FindFirstValue(ClaimTypes.NameIdentifier)!);
        
        var orderId = int.Parse(context.RouteData.Values["orderId"]?.ToString()!);
        
        var orderRepo = context.HttpContext.RequestServices
            .GetRequiredService<IOrderRepository>();
        
        var order = await orderRepo.GetByIdAsync(orderId);
        
        if (order is null)
        {
            context.Result = new NotFoundResult();
            return;
        }
        
        if (order.CustomerId != userId && !context.HttpContext.User.IsInRole("Admin"))
        {
            context.Result = new ForbidResult();
        }
    }
}

[RequiresOrderOwnership]
[HttpGet("orders/{orderId}")]
public async Task<IActionResult> GetOrder(int orderId) { ... }
```
