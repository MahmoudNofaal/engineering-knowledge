## 90.01 — Window Functions

**Ranking functions**
```sql
SELECT
    EmployeeId,
    Name,
    Department,
    Salary,
    
    -- ROW_NUMBER: unique sequential number per partition
    ROW_NUMBER() OVER (PARTITION BY Department ORDER BY Salary DESC) AS RowNum,
    
    -- RANK: gaps in rank for ties
    RANK() OVER (PARTITION BY Department ORDER BY Salary DESC) AS Rank,
    -- Example: 1, 2, 2, 4 (rank 3 skipped because two tied at 2)
    
    -- DENSE_RANK: no gaps for ties
    DENSE_RANK() OVER (PARTITION BY Department ORDER BY Salary DESC) AS DenseRank,
    -- Example: 1, 2, 2, 3
    
    -- NTILE: divide into N buckets
    NTILE(4) OVER (ORDER BY Salary DESC) AS Quartile  -- 1=top 25%, 4=bottom 25%

FROM Employees;

-- Get top earner per department
WITH Ranked AS (
    SELECT *,
        ROW_NUMBER() OVER (PARTITION BY Department ORDER BY Salary DESC) AS Rn
    FROM Employees
)
SELECT EmployeeId, Name, Department, Salary
FROM Ranked
WHERE Rn = 1;
```

**Aggregate window functions**
```sql
SELECT
    OrderId,
    CustomerId,
    Total,
    OrderDate,
    
    -- Running total
    SUM(Total) OVER (PARTITION BY CustomerId ORDER BY OrderDate) AS RunningTotal,
    
    -- Average over all rows for this customer
    AVG(Total) OVER (PARTITION BY CustomerId) AS CustomerAvgOrderValue,
    
    -- Row compared to partition
    Total / SUM(Total) OVER (PARTITION BY CustomerId) * 100 AS PctOfCustomerTotal,
    
    -- Moving average (last 3 orders)
    AVG(Total) OVER (
        PARTITION BY CustomerId
        ORDER BY OrderDate
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) AS MovingAvg3,
    
    -- Total across all
    SUM(Total) OVER () AS GrandTotal,
    
    -- Count of orders in window
    COUNT(*) OVER (PARTITION BY CustomerId) AS CustomerOrderCount

FROM Orders;
```

**LAG and LEAD — access adjacent rows**
```sql
SELECT
    OrderDate,
    Total,
    
    -- Previous row in partition
    LAG(Total, 1, 0) OVER (ORDER BY OrderDate) AS PreviousTotal,
    
    -- Next row in partition
    LEAD(Total, 1, 0) OVER (ORDER BY OrderDate) AS NextTotal,
    
    -- Change from previous
    Total - LAG(Total, 1, 0) OVER (ORDER BY OrderDate) AS ChangeFromPrevious,
    
    -- Percentage change
    CASE
        WHEN LAG(Total) OVER (ORDER BY OrderDate) = 0 THEN NULL
        ELSE (Total - LAG(Total) OVER (ORDER BY OrderDate)) /
             LAG(Total) OVER (ORDER BY OrderDate) * 100
    END AS PctChange

FROM Orders
ORDER BY OrderDate;
```

**FIRST_VALUE and LAST_VALUE**
```sql
SELECT
    Department,
    EmployeeName,
    Salary,
    
    -- Highest salary in department
    FIRST_VALUE(EmployeeName) OVER (
        PARTITION BY Department ORDER BY Salary DESC
        ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
    ) AS TopEarnerName,
    
    -- Lowest salary in department
    LAST_VALUE(EmployeeName) OVER (
        PARTITION BY Department ORDER BY Salary DESC
        ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
    ) AS BottomEarnerName

FROM Employees;
```

## 90.02 — CTEs (Common Table Expressions) — All Forms

**Recursive CTE — tree traversal**
```sql
-- Build org chart (employee hierarchy)
WITH OrgChart AS (
    -- Anchor: top-level employees
    SELECT EmployeeId, Name, ManagerId, Name AS Path, 0 AS Level
    FROM Employees
    WHERE ManagerId IS NULL
    
    UNION ALL
    
    -- Recursive: join to find subordinates
    SELECT e.EmployeeId, e.Name, e.ManagerId,
           CAST(oc.Path + ' > ' + e.Name AS NVARCHAR(MAX)),
           oc.Level + 1
    FROM Employees e
    INNER JOIN OrgChart oc ON e.ManagerId = oc.EmployeeId
)
SELECT EmployeeId, Name, Level, Path
FROM OrgChart
ORDER BY Level, Path;
-- MAXRECURSION hint if deep hierarchy: OPTION (MAXRECURSION 100)

-- Category tree
WITH CategoryTree AS (
    SELECT CategoryId, Name, ParentId, CAST(Name AS NVARCHAR(MAX)) AS FullPath
    FROM Categories
    WHERE ParentId IS NULL
    
    UNION ALL
    
    SELECT c.CategoryId, c.Name, c.ParentId, ct.FullPath + ' > ' + c.Name
    FROM Categories c
    INNER JOIN CategoryTree ct ON c.ParentId = ct.CategoryId
)
SELECT * FROM CategoryTree;
```

**CTE for complex queries**
```sql
-- Multiple CTEs for readable complex query
WITH
MonthlyRevenue AS (
    SELECT
        YEAR(OrderDate) AS Year,
        MONTH(OrderDate) AS Month,
        SUM(Total) AS Revenue,
        COUNT(*) AS OrderCount
    FROM Orders
    WHERE Status != 'Cancelled'
    GROUP BY YEAR(OrderDate), MONTH(OrderDate)
),
RevenueWithGrowth AS (
    SELECT
        Year, Month, Revenue, OrderCount,
        LAG(Revenue) OVER (ORDER BY Year, Month) AS PreviousRevenue,
        Revenue - LAG(Revenue) OVER (ORDER BY Year, Month) AS Growth
    FROM MonthlyRevenue
),
TopProducts AS (
    SELECT TOP 5
        p.Name,
        SUM(oi.Quantity * oi.UnitPrice) AS Revenue
    FROM OrderItems oi
    JOIN Products p ON oi.ProductId = p.Id
    JOIN Orders o ON oi.OrderId = o.Id
    WHERE o.Status != 'Cancelled'
    GROUP BY p.Name
    ORDER BY Revenue DESC
)
SELECT rwg.*, tp.Name AS TopProduct
FROM RevenueWithGrowth rwg
CROSS JOIN (SELECT TOP 1 Name FROM TopProducts) tp
ORDER BY rwg.Year, rwg.Month;
```

## 90.03 — Index Design Strategy

**Covering index**
```sql
-- Query: find active orders with customer name
SELECT o.Id, o.Status, o.Total, c.Name
FROM Orders o
JOIN Customers c ON o.CustomerId = c.Id
WHERE o.Status = 'Active'
ORDER BY o.CreatedAt DESC;

-- Index covering all needed columns for Orders table scan
CREATE NONCLUSTERED INDEX IX_Orders_Status_CreatedAt
ON Orders (Status, CreatedAt DESC)
INCLUDE (Total, CustomerId);  -- INCLUDE: available in index leaf pages
-- Index contains: Status (search), CreatedAt (sort), Total and CustomerId (avoid table lookup)

-- For Customers: need Name by Id (default clustered PK covers this)
```

**Index usage analysis**
```sql
-- Find missing indexes suggested by query optimizer
SELECT
    DB_NAME() AS DatabaseName,
    OBJECT_NAME(d.object_id) AS TableName,
    d.equality_columns,
    d.inequality_columns,
    d.included_columns,
    s.avg_total_user_cost * s.avg_user_impact / 100 * (s.user_seeks + s.user_scans) AS Score
FROM sys.dm_db_missing_index_details d
JOIN sys.dm_db_missing_index_groups g ON d.index_handle = g.index_handle
JOIN sys.dm_db_missing_index_group_stats s ON g.index_group_handle = s.group_handle
WHERE DB_ID() = d.database_id
ORDER BY Score DESC;

-- Find unused indexes (waste of write overhead)
SELECT
    OBJECT_NAME(i.object_id) AS TableName,
    i.name AS IndexName,
    i.type_desc,
    s.user_seeks, s.user_scans, s.user_lookups, s.user_updates
FROM sys.indexes i
LEFT JOIN sys.dm_db_index_usage_stats s
    ON i.object_id = s.object_id AND i.index_id = s.index_id
    AND s.database_id = DB_ID()
WHERE OBJECT_SCHEMA_NAME(i.object_id) != 'sys'
    AND i.type_desc != 'HEAP'
    AND (s.user_seeks + s.user_scans + s.user_lookups) = 0
    AND i.is_primary_key = 0
    AND i.is_unique_constraint = 0
ORDER BY s.user_updates DESC;
-- Indexes with many updates but zero reads → candidates for dropping
```
