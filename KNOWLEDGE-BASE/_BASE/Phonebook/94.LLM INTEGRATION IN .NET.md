## 94.01 — Microsoft.Extensions.AI (Unified AI Abstraction)

**What it is**
Microsoft's abstraction layer for AI models. Write code once, swap providers (OpenAI, Azure OpenAI, local Ollama, etc.)

```csharp
// NuGet: Microsoft.Extensions.AI, Microsoft.Extensions.AI.OpenAI

// Setup
builder.Services.AddChatClient(new OpenAIClient("your-api-key")
    .GetChatClient("gpt-4o-mini"))
    .UseFunctionInvocation()       // Enable tool calling
    .UseDistributedCache()         // Cache responses
    .UseLogging()                  // Log requests/responses
    .UseOpenTelemetry();           // Trace AI calls

// OR: Local model via Ollama
builder.Services.AddChatClient(new OllamaChatClient("http://localhost:11434", "llama3.2"))
    .UseLogging();

// Basic chat
public class ChatService
{
    private readonly IChatClient _chat;
    
    public ChatService(IChatClient chat) => _chat = chat;
    
    public async Task<string> GetResponseAsync(string userMessage, CancellationToken ct)
    {
        var response = await _chat.CompleteAsync(userMessage, ct);
        return response.Message.Text ?? "";
    }
    
    // With conversation history
    public async Task<string> ContinueConversationAsync(
        List<ChatMessage> history,
        string newUserMessage,
        CancellationToken ct)
    {
        history.Add(new ChatMessage(ChatRole.User, newUserMessage));
        
        var response = await _chat.CompleteAsync(history, ct);
        history.Add(response.Message);
        
        return response.Message.Text ?? "";
    }
    
    // Streaming response (token by token)
    public async IAsyncEnumerable<string> StreamResponseAsync(
        string userMessage,
        [EnumeratorCancellation] CancellationToken ct = default)
    {
        await foreach (var update in _chat.CompleteStreamingAsync(userMessage, ct))
        {
            if (update.Text is not null)
                yield return update.Text;
        }
    }
}
```

**Tool calling (function calling)**
```csharp
// Define tools that the LLM can call
[Description("Get the current weather for a city")]
private static async Task<WeatherResult> GetWeatherAsync(
    [Description("The city name")] string city,
    [Description("Units: 'celsius' or 'fahrenheit'")] string units = "celsius")

    Tools = [
```

**Embeddings for semantic search**
```csharp
// NuGet: Microsoft.Extensions.AI, Microsoft.Extensions.AI.OpenAI

builder.Services.AddEmbeddingGenerator<string, Embedding<float>>(
    new OpenAIClient("api-key").GetEmbeddingClient("text-embedding-3-small"));

// Generate embedding
public class SemanticSearchService
{
    private readonly IEmbeddingGenerator<string, Embedding<float>> _embeddings;
    private readonly IVectorStore _vectorStore;
    
    public async Task IndexDocumentAsync(Document doc, CancellationToken ct)
    {
        // Convert text to vector
        var embedding = await _embeddings.GenerateEmbeddingVectorAsync(doc.Content, ct);
        
        await _vectorStore.UpsertAsync(new VectorRecord
        {
            Id = doc.Id.ToString(),
            Vector = embedding.ToArray(),
            Metadata = new { doc.Title, doc.Content }
        }, ct);
    }
    
    public async Task<List<Document>> SearchAsync(string query, int topK, CancellationToken ct)
    {
        // Convert query to vector
        var queryVector = await _embeddings.GenerateEmbeddingVectorAsync(query, ct);
        
        // Find nearest neighbors
        var results = await _vectorStore.SearchAsync(queryVector.ToArray(), topK, ct);
        
        return results.Select(r => r.ToDocument()).ToList();
    }
}
```

## 94.02 — Retrieval-Augmented Generation (RAG)

**What it is**
Augment LLM responses with your own documents. LLM generates answers based on retrieved context, not just training data.

```csharp
// Complete RAG pipeline
public class RagService
{
    private readonly IChatClient _chat;
    private readonly IEmbeddingGenerator<string, Embedding<float>> _embeddings;
    private readonly IVectorStore _vectorStore;
    
    // Step 1: Index documents
    public async Task IndexAsync(IEnumerable<Document> docs, CancellationToken ct)
    {
        foreach (var doc in docs)
        {
            // Chunk large documents
            var chunks = ChunkText(doc.Content, maxTokens: 500, overlap: 50);
            
            foreach (var (chunk, index) in chunks.Select((c, i) => (c, i)))
            {
                var embedding = await _embeddings.GenerateEmbeddingVectorAsync(chunk, ct);
                
                await _vectorStore.UpsertAsync(new VectorRecord
                {
                    Id = $"{doc.Id}:{index}",
                    Vector = embedding.ToArray(),
                    Metadata = new
                    {
                        doc.Id,
                        doc.Title,
                        ChunkIndex = index,
                        Content = chunk
                    }
                }, ct);
            }
        }
    }
    
    // Step 2: Answer with RAG
    public async Task<string> AskAsync(string question, CancellationToken ct)
    {
        // Retrieve relevant context
        var questionEmbedding = await _embeddings.GenerateEmbeddingVectorAsync(question, ct);
        var relevantChunks = await _vectorStore.SearchAsync(questionEmbedding.ToArray(), topK: 5, ct);
        
        // Build context
        var context = string.Join("\n\n---\n\n",
            relevantChunks.Select(r => $"Source: {r.Metadata.Title}\n{r.Metadata.Content}"));
        
        // Build prompt with context
        var systemPrompt = $"""
            You are a helpful assistant. Answer questions based ONLY on the provided context.
            If the answer is not in the context, say "I don't have information about that."
            
            Context:
            {context}
            """;
        
        var messages = new List<ChatMessage>
        {
            new(ChatRole.System, systemPrompt),
            new(ChatRole.User, question)
        };
        
        var response = await _chat.CompleteAsync(messages, ct);
        return response.Message.Text ?? "";
    }
    
    // Text chunking strategy
    private static List<string> ChunkText(string text, int maxTokens, int overlap)
    {
        // Approximate: 1 token ≈ 4 characters
        int maxChars = maxTokens * 4;
        int overlapChars = overlap * 4;
        
        var chunks = new List<string>();
        int start = 0;
        
        while (start < text.Length)
        {
            int end = Math.Min(start + maxChars, text.Length);
            
            // Try to break at sentence boundary
            if (end < text.Length)
            {
                int sentenceEnd = text.LastIndexOf('.', end, Math.Min(100, end - start));
                if (sentenceEnd > start) end = sentenceEnd + 1;
            }
            
            chunks.Add(text[start..end]);
            start = end - overlapChars;  // Overlap preserves context
            if (start >= text.Length) break;
        }
        
        return chunks;
    }
}
```

**Prompt engineering patterns**
```csharp
// Chain of thought — ask the model to reason step by step
var prompt = """
    Analyze this order and determine if it should be flagged for fraud review.
    
    Order details:
    - Amount: $2,499
    - Customer location: Egypt
    - Shipping to: Russia
    - Payment method: Gift card
    - Account age: 2 days
    
    Think step by step:
    1. What risk factors are present?
    2. What is the likelihood of fraud?
    3. What is your recommendation?
    
    Then provide your final answer as: APPROVE, REVIEW, or REJECT
    """;

// Few-shot prompting — provide examples
var prompt = """
    Classify the following customer support messages into categories:
    BILLING, TECHNICAL, GENERAL, URGENT
    
    Examples:
    Message: "I was charged twice for my order" → BILLING
    Message: "The app crashes when I open it" → TECHNICAL
    Message: "When will my order arrive?" → GENERAL
    Message: "My account was hacked!" → URGENT
    
    Now classify these:
    Message: "I can't log in to my account" → ?
    Message: "I need a refund immediately" → ?
    """;

// Structured output (JSON mode)
var options = new ChatOptions
{
    ResponseFormat = ChatResponseFormat.Json
```
