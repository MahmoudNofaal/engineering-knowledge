## 33.01 — The Specification Pattern

**What it is**
Encapsulate query predicates as composable, reusable, testable objects. Prevents complex query logic from leaking into handlers.

**Complete implementation**
```csharp
public abstract class Specification<T>
{
    public abstract Expression<Func<T, bool>> Criteria { get; }
    
    public List<Expression<Func<T, object>>> Includes { get; } = new();
    public List<string> IncludeStrings { get; } = new();
    public Expression<Func<T, object>>? OrderBy { get; protected set; }
    public Expression<Func<T, object>>? OrderByDescending { get; protected set; }
    public int? Take { get; protected set; }
    public int? Skip { get; protected set; }
    
    protected void AddInclude(Expression<Func<T, object>> include)
        => Includes.Add(include);
    
    protected void ApplyOrderBy(Expression<Func<T, object>> orderByExpression)
        => OrderBy = orderByExpression;
    
    protected void ApplyPaging(int pageIndex, int pageSize)
    {
        Skip = pageSize * (pageIndex - 1);
        Take = pageSize;
    }
    
    public Specification<T> And(Specification<T> other)
        => new AndSpecification<T>(this, other);
    
    public Specification<T> Or(Specification<T> other)
        => new OrSpecification<T>(this, other);
}

// Example specifications
public class ActiveOrdersSpec : Specification<Order>
{
    public override Expression<Func<Order, bool>> Criteria =>
        o => o.Status != OrderStatus.Cancelled && o.Status != OrderStatus.Completed;
}

public class OrdersByCustomerSpec : Specification<Order>
{
    private readonly int _customerId;
    
    public OrdersByCustomerSpec(int customerId)
    {
        _customerId = customerId;
        AddInclude(o => o.Items);
        ApplyOrderBy(o => o.CreatedAt);
    }
    
    public override Expression<Func<Order, bool>> Criteria =>
        o => o.CustomerId == _customerId;
}

public class HighValueOrdersSpec : Specification<Order>
{
    private readonly decimal _threshold;
    
    public HighValueOrdersSpec(decimal threshold) => _threshold = threshold;
    
    public override Expression<Func<Order, bool>> Criteria =>
        o => o.Total >= _threshold;
}

// Generic repository using specifications
public class Repository<T> where T : class
{
    private readonly AppDbContext _db;
    
    public async Task<List<T>> FindAsync(Specification<T> spec, CancellationToken ct = default)
    {
        var query = ApplySpec(_db.Set<T>().AsQueryable(), spec);
        return await query.ToListAsync(ct);
    }
    
    private IQueryable<T> ApplySpec(IQueryable<T> query, Specification<T> spec)
    {
        query = query.Where(spec.Criteria);
        
        query = spec.Includes.Aggregate(query,
            (current, include) => current.Include(include));
        
        if (spec.OrderBy is not null)
            query = query.OrderBy(spec.OrderBy);
        
        if (spec.Skip.HasValue) query = query.Skip(spec.Skip.Value);
        if (spec.Take.HasValue) query = query.Take(spec.Take.Value);
        
        return query;
    }
}

// Usage
var spec = new OrdersByCustomerSpec(customerId)
    .And(new ActiveOrdersSpec())
    .And(new HighValueOrdersSpec(500));

var orders = await _repository.FindAsync(spec);
```

## 33.02 — The Unit of Work Pattern

**What it is**
Wraps a set of operations into a single transaction. Ensures either all operations succeed or none do.

```csharp
public interface IUnitOfWork : IAsyncDisposable
{
    IOrderRepository Orders { get; }
    IProductRepository Products { get; }
    ICustomerRepository Customers { get; }
    
    Task<int> SaveChangesAsync(CancellationToken ct = default);
    Task BeginTransactionAsync(CancellationToken ct = default);
    Task CommitAsync(CancellationToken ct = default);
    Task RollbackAsync(CancellationToken ct = default);
}

public class UnitOfWork : IUnitOfWork
{
    private readonly AppDbContext _db;
    private IDbContextTransaction? _transaction;
    
    public IOrderRepository Orders { get; }
    public IProductRepository Products { get; }
    public ICustomerRepository Customers { get; }
    
    public UnitOfWork(AppDbContext db, IOrderRepository orders,
        IProductRepository products, ICustomerRepository customers)
    {
        _db = db;
        Orders = orders;
        Products = products;
        Customers = customers;
    }
    
    public async Task<int> SaveChangesAsync(CancellationToken ct = default)
    {
        // Dispatch domain events before saving
        var aggregates = _db.ChangeTracker.Entries<AggregateRoot>()
            .Where(e => e.Entity.GetDomainEvents().Any())
            .Select(e => e.Entity)
            .ToList();
        
        var events = aggregates.SelectMany(a => a.GetDomainEvents()).ToList();
        aggregates.ForEach(a => a.ClearDomainEvents());
        
        var result = await _db.SaveChangesAsync(ct);
        
        // Publish domain events after successful save
        foreach (var @event in events)
            await _mediator.Publish(@event, ct);
        
        return result;
    }
    
    public async Task BeginTransactionAsync(CancellationToken ct = default)
        => _transaction = await _db.Database.BeginTransactionAsync(ct);
    
    public async Task CommitAsync(CancellationToken ct = default)
    {
        await _db.SaveChangesAsync(ct);
        await _transaction!.CommitAsync(ct);
    }
    
    public async Task RollbackAsync(CancellationToken ct = default)
        => await _transaction!.RollbackAsync(ct);
    
    public async ValueTask DisposeAsync()
    {
        if (_transaction is not null) await _transaction.DisposeAsync();
        await _db.DisposeAsync();
    }
}
```

## 33.03 — Multi-Tenancy Patterns

**What it is**
A single application instance serves multiple tenants (clients), with data isolation between them.

**Database-per-tenant**
Each tenant has their own database. Strongest isolation.
```csharp
public class TenantDbContextFactory
{
    private readonly ITenantResolver _tenantResolver;
    private readonly IConfiguration _config;
    
    public AppDbContext Create()
    {
        var tenantId = _tenantResolver.GetCurrentTenantId();
        var connectionString = _config[$"Tenants:{tenantId}:ConnectionString"];
        
        var options = new DbContextOptionsBuilder<AppDbContext>()
            .UseSqlServer(connectionString)
            .Options;
        
        return new AppDbContext(options);
    }
}
```

**Schema-per-tenant (same database)**
```sql
-- Tenant A data:
CREATE SCHEMA tenant_a;
CREATE TABLE tenant_a.Orders (...);

-- Tenant B data:
CREATE SCHEMA tenant_b;
CREATE TABLE tenant_b.Orders (...);
```

**Row-level isolation (shared tables)**
All tenants share tables. TenantId column on every table.
```csharp
// Global query filter automatically adds WHERE TenantId = @tenantId
protected override void OnModelCreating(ModelBuilder modelBuilder)
{
    modelBuilder.Entity<Order>()
        .HasQueryFilter(o => o.TenantId == _currentTenant.TenantId);
    
    modelBuilder.Entity<Product>()
        .HasQueryFilter(p => p.TenantId == _currentTenant.TenantId);
}

// Scoped service to get current tenant
public interface ICurrentTenant
{
    int TenantId { get; }
}

public class CurrentTenant : ICurrentTenant
{
    private readonly IHttpContextAccessor _accessor;
    
    public int TenantId
    {
        get
        {
            var claim = _accessor.HttpContext?.User.FindFirst("tenant_id");
            return int.Parse(claim?.Value ?? throw new InvalidOperationException("No tenant"));
        }
    }
}
```
