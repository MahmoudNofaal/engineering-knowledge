**Notification service**
```csharp
public class FcmNotificationService : INotificationService
{
    private readonly ILogger<FcmNotificationService> _logger;
    
    public async Task SendToDeviceAsync(
        string deviceToken,
        string title,
        string body,
        Dictionary<string, string>? data = null,
        CancellationToken ct = default)
    {
        var message = new Message
        {
            Token = deviceToken,
            Notification = new Notification { Title = title, Body = body },
            Data = data ?? new Dictionary<string, string>(),
            Android = new AndroidConfig
            {
                Priority = Priority.High,
                Notification = new AndroidNotification
                {
                    Sound = "default",
                    ClickAction = "FLUTTER_NOTIFICATION_CLICK"
                }
            },
            Apns = new ApnsConfig
            {
                Headers = new Dictionary<string, string>
                {
                    ["apns-priority"] = "10"
                },
                Aps = new Aps
                {
                    Alert = new ApsAlert { Title = title, Body = body },
                    Sound = "default",
                    Badge = 1
                }
            }
        };
        
        try
        {
            var response = await FirebaseMessaging.DefaultInstance.SendAsync(message, ct);
            _logger.LogInformation("FCM sent: {MessageId}", response);
        }
        catch (FirebaseMessagingException ex)
        {
            if (ex.MessagingErrorCode == MessagingErrorCode.Unregistered ||
                ex.MessagingErrorCode == MessagingErrorCode.InvalidRegistrationToken)
            {
                // Token is invalid — remove from database
                _logger.LogWarning("FCM token invalid for device: {Token}", deviceToken[..10] + "...");
                throw new InvalidDeviceTokenException(deviceToken);
            }
            throw;
        }
    }
    
    public async Task SendToTopicAsync(
        string topic,
        string title,
        string body,
        Dictionary<string, string>? data = null,
        CancellationToken ct = default)
    {
        var message = new Message
        {
            Topic = topic,  // All devices subscribed to this topic
            Notification = new Notification { Title = title, Body = body },
            Data = data
        };
        
        var response = await FirebaseMessaging.DefaultInstance.SendAsync(message, ct);
        _logger.LogInformation("FCM topic message sent: {MessageId}", response);
    }
    
    public async Task SendBatchAsync(
        IEnumerable<string> deviceTokens,
        string title,
        string body,
        Dictionary<string, string>? data = null,
        CancellationToken ct = default)
    {
        var tokens = deviceTokens.ToList();
        
        // FCM allows up to 500 tokens per batch
        foreach (var chunk in tokens.Chunk(500))
        {
            var multicastMessage = new MulticastMessage
            {
                Tokens = chunk,
                Notification = new Notification { Title = title, Body = body },
                Data = data
            };
            
            var response = await FirebaseMessaging.DefaultInstance
                .SendEachForMulticastAsync(multicastMessage, ct);
            
            _logger.LogInformation(
                "FCM batch: {Success} succeeded, {Failure} failed out of {Total}",
                response.SuccessCount, response.FailureCount, chunk.Length);
            
            // Handle failed tokens
            for (int i = 0; i < response.Responses.Count; i++)
            {
                if (!response.Responses[i].IsSuccess)
                    await HandleFailedTokenAsync(chunk[i], response.Responses[i].Exception, ct);
            }
        }
    }
    
    private async Task HandleFailedTokenAsync(
        string token, FirebaseException exception, CancellationToken ct)
    {
        if (exception is FirebaseMessagingException fcmEx &&
            (fcmEx.MessagingErrorCode == MessagingErrorCode.Unregistered ||
             fcmEx.MessagingErrorCode == MessagingErrorCode.InvalidRegistrationToken))
        {
            // Remove invalid token from database
            await _deviceTokenRepository.RemoveAsync(token, ct);
        }
    }
}
```

**Device token management**
```csharp
// Entity
public class DeviceToken
{
    public int Id { get; private set; }
    public int UserId { get; private set; }
    public string Token { get; private set; } = null!;
    public string Platform { get; private set; } = null!;  // "android", "ios", "web"
    public DateTime RegisteredAt { get; private set; }
    public DateTime? LastUsedAt { get; private set; }
    public bool IsActive { get; private set; }
    
    public static DeviceToken Register(int userId, string token, string platform)
        => new() { UserId = userId, Token = token, Platform = platform,
                   RegisteredAt = DateTime.UtcNow, IsActive = true };
    
    public void Deactivate() => IsActive = false;
}

// Endpoint to register device token
app.MapPost("/api/devices/register", async (
    RegisterDeviceTokenRequest request,
    ICurrentUser currentUser,
    IDeviceTokenRepository tokenRepo,
    CancellationToken ct) =>
{
    // Upsert: add if new, update timestamp if exists
    var existing = await tokenRepo.GetByTokenAsync(request.Token, ct);
    if (existing is not null)
    {
        existing.RecordUsage();
        await tokenRepo.UpdateAsync(existing, ct);
    }
    else
    {
        var deviceToken = DeviceToken.Register(currentUser.UserId, request.Token, request.Platform);
        await tokenRepo.AddAsync(deviceToken, ct);
    }
    
    return Results.Ok();
})
.RequireAuthorization();
```

**Hangfire job for notifications**
```csharp
// Using Hangfire to send notifications asynchronously
// The API returns immediately; the notification is sent in the background

public class OrderService
{
    private readonly IBackgroundJobClient _jobs;
    
    public async Task<Order> CreateOrderAsync(CreateOrderCommand cmd, CancellationToken ct)
    {
        var order = /* ... create order ... */;
        
        // Schedule notification as background job — not blocking the response
        _jobs.Enqueue<INotificationJob>(
            j => j.SendOrderCreatedNotificationAsync(order.Id, CancellationToken.None));
        
        return order;
    }
}

public class NotificationJob : INotificationJob
{
    private readonly INotificationService _fcm;
    private readonly IOrderRepository _orders;
    private readonly IDeviceTokenRepository _tokens;
    
    public async Task SendOrderCreatedNotificationAsync(int orderId, CancellationToken ct)
    {
        var order = await _orders.GetByIdAsync(orderId, ct);
        var deviceTokens = await _tokens.GetActiveByUserIdAsync(order!.CustomerId, ct);
        
        if (!deviceTokens.Any()) return;
        
        await _fcm.SendBatchAsync(
            deviceTokens.Select(dt => dt.Token),
            title: "Order Confirmed! 🎉",
            body: $"Your order #{orderId} has been confirmed. Total: {order.Total:C}",
            data: new Dictionary<string, string>
            {
                ["type"] = "order_created",
                ["orderId"] = orderId.ToString(),
                ["screen"] = "OrderDetails"
            },
            ct);
    }
}
```
