## 75.01 — ASP.NET Core Rate Limiter (.NET 7+)

**Fixed window**
```csharp
builder.Services.AddRateLimiter(options =>
{
    options.AddFixedWindowLimiter("fixed", config =>
    {
        config.Window = TimeSpan.FromMinutes(1);
        config.PermitLimit = 100;           // 100 requests per window
        config.QueueLimit = 20;             // Queue up to 20 excess requests
        config.QueueProcessingOrder = QueueProcessingOrder.OldestFirst;
    });
    
    options.RejectionStatusCode = StatusCodes.Status429TooManyRequests;
    
    options.OnRejected = async (context, ct) =>
    {
        context.HttpContext.Response.Headers.RetryAfter =
            DateTimeOffset.UtcNow.AddMinutes(1).ToString("R");
        await context.HttpContext.Response.WriteAsync(
            "Too many requests. Please try again later.", ct);
    };
});
```

**Sliding window**
```csharp
options.AddSlidingWindowLimiter("sliding", config =>
{
    config.Window = TimeSpan.FromMinutes(1);
    config.SegmentsPerWindow = 6;           // 10-second segments
    config.PermitLimit = 100;
    // Sliding: counts requests in the last 60 seconds, not fixed 60-second blocks
    // Less bursty than fixed window
});
```

**Token bucket (burst + sustained rate)**
```csharp
options.AddTokenBucketLimiter("token-bucket", config =>
{
    config.TokenLimit = 20;                 // Max burst size
    config.TokensPerPeriod = 5;             // Replenishment rate
    config.ReplenishmentPeriod = TimeSpan.FromSeconds(1);  // Per second
    config.AutoReplenishment = true;
    // Allows bursting up to 20, sustained rate of 5/sec
});
```

**Per-user/per-IP rate limiting**
```csharp
options.AddPolicy("per-user", httpContext =>
{
    // Rate limit based on user ID if authenticated, IP if not
    var userId = httpContext.User.FindFirstValue(ClaimTypes.NameIdentifier);
    
    if (userId is not null)
    {
        return RateLimitPartition.GetFixedWindowLimiter(
            partitionKey: $"user:{userId}",
            factory: _ => new FixedWindowRateLimiterOptions
            {
                Window = TimeSpan.FromMinutes(1),
                PermitLimit = 200  // Authenticated: 200 req/min
            });
    }
    
    var ip = httpContext.Connection.RemoteIpAddress?.ToString() ?? "unknown";
    return RateLimitPartition.GetFixedWindowLimiter(
        partitionKey: $"ip:{ip}",
        factory: _ => new FixedWindowRateLimiterOptions
        {
            Window = TimeSpan.FromMinutes(1),
            PermitLimit = 30    // Anonymous: 30 req/min
        });
});

// Apply to endpoints
app.MapGet("/api/data", GetData)
    .RequireRateLimiting("per-user");

// Or global middleware (after routing, before authorization)
app.UseRateLimiter();
```
