## 98.01 — BenchmarkDotNet

**Setup and patterns**
```csharp
// NuGet: BenchmarkDotNet

[MemoryDiagnoser]          // Tracks allocations
[SimpleJob(RuntimeMoniker.Net80)]
[Orderer(SummaryOrderPolicy.FastestToSlowest)]
public class StringConcatenationBenchmark
{
    private readonly string[] _parts;
    
    public StringConcatenationBenchmark()
        => _parts = Enumerable.Range(0, 100).Select(i => i.ToString()).ToArray();
    
    [Benchmark(Baseline = true)]
    public string StringConcat()
    {
        var result = "";
        foreach (var part in _parts)
            result += part;
        return result;
    }
    
    [Benchmark]
    public string StringBuilder()
    {
        var sb = new StringBuilder();
        foreach (var part in _parts)
            sb.Append(part);
        return sb.ToString();
    }
    
    [Benchmark]
    public string StringJoin() => string.Join("", _parts);
    
    [Benchmark]
    public string StringCreate()
    {
        var totalLength = _parts.Sum(p => p.Length);
        return string.Create(totalLength, _parts, (span, parts) =>
        {
            int pos = 0;
            foreach (var part in parts)
            {
                part.AsSpan().CopyTo(span[pos..]);
                pos += part.Length;
            }
        });
    }
}

// Run benchmarks
// dotnet run -c Release -- --filter '*StringConcatenation*'

// Typical results:
// | Method        | Mean      | Allocated |
// |---------------|-----------|-----------|
// | StringConcat  | 12,450 ns | 25,600 B  |
// | StringBuilder | 1,234 ns  | 512 B     |
// | StringJoin    | 890 ns    | 256 B     |
// | StringCreate  | 456 ns    | 128 B     |
```

**LINQ vs. manual loop benchmark**
```csharp
[MemoryDiagnoser]
public class LinqVsLoop
{
    private readonly int[] _data = Enumerable.Range(0, 10000).ToArray();
    
    [Benchmark(Baseline = true)]
    public long LinqSum() => _data.Where(x => x % 2 == 0).Sum(x => (long)x);
    
    [Benchmark]
    public long ManualLoop()
    {
        long sum = 0;
        foreach (var x in _data)
            if (x % 2 == 0)
                sum += x;
        return sum;
    }
    
    [Benchmark]
    public long SpanLoop()
    {
        long sum = 0;
        var span = _data.AsSpan();
        for (int i = 0; i < span.Length; i++)
            if (span[i] % 2 == 0)
                sum += span[i];
        return sum;
    }
}
// SpanLoop: fastest, zero allocation
// ManualLoop: nearly as fast, zero allocation
// LinqSum: slowest, allocates enumerators
```
